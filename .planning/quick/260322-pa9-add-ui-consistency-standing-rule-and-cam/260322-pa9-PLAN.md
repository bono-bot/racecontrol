```javascript
function navigateFullscreen(direction) {
  if (!activeCamera || cameras.length === 0) return;
  var navigable = cameras.filter(function(c) {
    return c.nvr_channel != null && c.status !== 'offline' && c.status !== 'disconnected';
  });
  if (navigable.length === 0) return;
  var curIdx = navigable.findIndex(function(c) { return c.nvr_channel === activeCamera.nvr_channel; });
  if (curIdx === -1) curIdx = 0;
  var nextIdx = (curIdx + direction + navigable.length) % navigable.length;
  openFullscreen(navigable[nextIdx]);
}
```

6. Add `handleGridKeydown(e)` function (place after `navigateFullscreen`):

```javascript
function handleGridKeydown(e) {
  var tiles = Array.from(gridEl.querySelectorAll('.cam'));
  if (tiles.length === 0) return;

  if (e.key === 'Enter' && focusedTileIndex >= 0 && focusedTileIndex < tiles.length) {
    e.preventDefault();
    var ch = parseInt(tiles[focusedTileIndex].dataset.channel);
    var cam = cameras.find(function(c) { return c.nvr_channel === ch; });
    if (cam && cam.nvr_channel != null && cam.status !== 'offline' && cam.status !== 'disconnected') {
      openFullscreen(cam);
    }
    return;
  }

  var cols = getGridColumns();
  var newIdx = focusedTileIndex;

  if (e.key === 'ArrowRight') { newIdx = focusedTileIndex + 1; }
  else if (e.key === 'ArrowLeft') { newIdx = focusedTileIndex - 1; }
  else if (e.key === 'ArrowDown') { newIdx = focusedTileIndex + cols; }
  else if (e.key === 'ArrowUp') { newIdx = focusedTileIndex - cols; }
  else { return; }

  e.preventDefault();
  if (focusedTileIndex < 0) { newIdx = 0; }
  if (newIdx < 0) newIdx = 0;
  if (newIdx >= tiles.length) newIdx = tiles.length - 1;

  if (focusedTileIndex >= 0 && focusedTileIndex < tiles.length) {
    tiles[focusedTileIndex].classList.remove('kbd-focus');
  }
  focusedTileIndex = newIdx;
  tiles[focusedTileIndex].classList.add('kbd-focus');
  tiles[focusedTileIndex].scrollIntoView({ block: 'nearest' });
}

function getGridColumns() {
  var w = window.innerWidth;
  if (w <= 500) return 1;
  if (w <= 800) return 2;
  if (w <= 1200) return 3;
  return 4;
}
```
