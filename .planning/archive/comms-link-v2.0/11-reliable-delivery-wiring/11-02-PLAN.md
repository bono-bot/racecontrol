7. **Add sendTaskRequest helper** inside wireBono() (or as a returned function):
```javascript
function sendTaskRequest(ws, payload, timeoutMs = TASK_TIMEOUT_MS) {
  const taskId = payload.taskId || randomUUID();
  const fullPayload = { ...payload, taskId };
  const raw = createMessage('task_request', 'bono', fullPayload);
  const parsed = JSON.parse(raw);
  if (ackTracker) {
    ackTracker.track(parsed.id, raw, 'task_request');
  } else {
    ws.send(raw);
  }
  const timer = setTimeout(() => {
    pendingTasks.delete(taskId);
    console.warn(`[TASK] Request ${taskId} timed out after ${timeoutMs}ms`);
  }, timeoutMs);
  pendingTasks.set(taskId, { timer, messageId: parsed.id });
  return taskId;
}
```
