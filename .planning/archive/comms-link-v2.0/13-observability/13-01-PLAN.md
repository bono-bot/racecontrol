```javascript
export class MetricsCollector {
  #startTime;
  #reconnectCount = 0;
  #latencies = [];
  #nowFn;
  #maxWindow = 100;

  constructor({ nowFn = Date.now } = {}) {
    this.#nowFn = nowFn;
    this.#startTime = this.#nowFn();
  }

  recordReconnect() { this.#reconnectCount++; }

  recordAckLatency(ms) {
    this.#latencies.push(ms);
    if (this.#latencies.length > this.#maxWindow) {
      this.#latencies.shift();
    }
  }

  snapshot() {
    const len = this.#latencies.length;
    let ackLatencyAvgMs = null;
    let ackLatencyP99Ms = null;
    if (len > 0) {
      ackLatencyAvgMs = Math.round(this.#latencies.reduce((a, b) => a + b, 0) / len);
      const sorted = [...this.#latencies].sort((a, b) => a - b);
      ackLatencyP99Ms = sorted[Math.floor(sorted.length * 0.99)];
    }
    return {
      uptimeMs: this.#nowFn() - this.#startTime,
      reconnectCount: this.#reconnectCount,
      ackLatencyAvgMs,
      ackLatencyP99Ms,
      ts: this.#nowFn(),
    };
  }
}
```
