```javascript
import { describe, it } from 'node:test';
import assert from 'node:assert/strict';
import { execFile } from 'node:child_process';
import { access, constants } from 'node:fs/promises';

describe('OBS-04: Email Fallback Validation', () => {
  const emailPath = process.env.SEND_EMAIL_PATH;

  it('SEND_EMAIL_PATH env var is set', { skip: !emailPath && 'SEND_EMAIL_PATH not set' }, () => {
    assert.ok(emailPath, 'SEND_EMAIL_PATH must point to send_email.js');
  });

  it('send_email.js file exists', { skip: !emailPath && 'SEND_EMAIL_PATH not set' }, async () => {
    await access(emailPath, constants.R_OK);
  });

  it('send_email.js executes without crash (dry-run)', { skip: !emailPath && 'SEND_EMAIL_PATH not set' }, (_, done) => {
    // Invoke with --help or minimal args to verify the script loads without error
    // If the script requires real args, this tests that it at least parses and runs
    execFile('node', [emailPath, '--help'], { timeout: 10000 }, (err, stdout, stderr) => {
      // Script may exit non-zero for --help, but should not throw/crash
      // We accept exit code 0 or 1 (help printed), but not timeout or signal
      if (err && err.killed) {
        assert.fail('send_email.js timed out -- script is broken');
      }
      // If we get here, the script at least loaded and ran
      done();
    });
  });

  it('E2E send test email to bono@racingpoint.in', {
    skip: (!emailPath && 'SEND_EMAIL_PATH not set') || (!process.env.EMAIL_E2E && 'Set EMAIL_E2E=1 to run live send'),
  }, (_, done) => {
    execFile('node', [
      emailPath,
      '--to', 'bono@racingpoint.in',
      '--subject', '[COMMS-LINK] E2E Fallback Validation',
      '--body', `Email fallback validated at ${new Date().toISOString()} from comms-link Phase 13`,
    ], { timeout: 30000 }, (err, stdout, stderr) => {
      if (err) {
        // Log detailed error for OAuth diagnosis
        console.error('[OBS-04] Email send failed:', stderr || err.message);
        // If this is an OAuth error, document it rather than fail hard
        if (stderr?.includes('getAuthClient') || stderr?.includes('refresh token')) {
          console.error('[OBS-04] KNOWN ISSUE: Gmail OAuth tokens expired. Email infrastructure code is correct; auth renewal is a separate ops task.');
          // Still fail the test to accurately report status
        }
        assert.fail(`send_email.js failed: ${err.message}`);
      }
      console.log('[OBS-04] Email sent successfully. Verify receipt at bono@racingpoint.in');
      done();
    });
  });
});
```

- Skip gracefully when SEND_EMAIL_PATH is not set (normal CI runs)
- Run the dry-run test to validate the script loads
- Only attempt live E2E send when EMAIL_E2E=1 is explicitly set
- Log diagnostic info if OAuth is broken (known issue from MEMORY.md)
