   ```javascript
   function sendExecRequest({ command, reason = '' }) {
     // Validate command exists in registry
     validateExecRequest({ command });
     const execId = `ex_${randomUUID().slice(0, 8)}`;
     const payload = { execId, command, reason, requestedBy: 'bono' };
     // Use sendTracked pattern (same as James side)
     const raw = createMessage('exec_request', 'bono', payload);
     const parsed = JSON.parse(raw);
     if (ackTracker) {
       ackTracker.track(parsed.id, raw, 'exec_request');
     } else {
       // Fallback: send directly to first connected client
       for (const ws of wss.clients) {
         if (ws.readyState === 1) ws.send(raw);
       }
     }
     return execId;
   }
   ```
