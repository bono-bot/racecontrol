    setAssist(podId: string, assistType: string, enabled: boolean): Promise<{ ok: boolean }> {
      return fetchApi(`/pods/${podId}/assists`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ assist_type: assistType, enabled }),
      });
    },

    setFfbGain(podId: string, percent: number): Promise<{ ok: boolean; ffb_percent?: number }> {
      return fetchApi(`/pods/${podId}/ffb`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ percent }),
      });
    },

    getAssistState(podId: string): Promise<{
      ok: boolean;
      abs?: number;
      tc?: number;
      auto_shifter?: boolean;
      ffb_percent?: number;
    }> {
      return fetchApi(`/pods/${podId}/assist-state`);
    },
    ```

      tc: number;        // 0=off, 1-4=level

      ffb_percent: number; // 10-100
    }
    ```

    ```tsx
    async function openSheet() {
      setSheetOpen(true);
      if (!reservation) return;
      try {
        const state = await api.getAssistState(reservation.pod_id);
        if (state.abs !== undefined) setAbsOn(state.abs > 0);
        if (state.tc !== undefined) setTcOn(state.tc > 0);
        if (state.auto_shifter !== undefined) setAutoTrans(state.auto_shifter);
        if (state.ffb_percent !== undefined) setFfbPercent(state.ffb_percent);
      } catch {
        // Use cached values if query fails
      }
    }
    ```

    Per user decision: "Drawer shows actual pod state when opened -- PWA queries rc-core for current assist/FFB values, not cached last-sent values."

    **3. Toggle handlers (instant send, no Apply button):**
    ```tsx
    async function toggleAssist(type: "abs" | "tc" | "transmission") {
      if (!reservation) return;
      const newValue = type === "abs" ? !absOn : type === "tc" ? !tcOn : !autoTrans;

      // Update local state immediately for responsive feel
      if (type === "abs") setAbsOn(newValue);
      else if (type === "tc") setTcOn(newValue);
      else setAutoTrans(newValue);

      try {
        await api.setAssist(reservation.pod_id, type, newValue);
        showConfirm(type === "abs" ? `ABS: ${newValue ? "ON" : "OFF"}` :
                     type === "tc" ? `TC: ${newValue ? "ON" : "OFF"}` :
                     `Transmission: ${newValue ? "AUTO" : "MANUAL"}`);
      } catch {
        // Revert on failure
        if (type === "abs") setAbsOn(!newValue);
        else if (type === "tc") setTcOn(!newValue);
        else setAutoTrans(!newValue);
      }
    }
    ```

    **4. FFB slider with 500ms debounce:**
    ```tsx
    function handleFfbChange(value: number) {
      setFfbPercent(value); // Update slider position immediately (visual)

      // Debounce the actual API call by 500ms
      if (ffbTimerRef.current) clearTimeout(ffbTimerRef.current);
      ffbTimerRef.current = setTimeout(async () => {
        if (!reservation) return;
        try {
          await api.setFfbGain(reservation.pod_id, value);
          showConfirm(`FFB: ${value}%`);
        } catch {
          // Slider already shows the value; no revert needed
        }
      }, 500);
    }
    ```

    Per user decision: "500ms debounce on slider changes -- only send HID command after 500ms of no movement to prevent flooding the wheelbase."

    function showConfirm(msg: string) {
