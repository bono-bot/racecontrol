    **Add new handler for SetAssist:**
    ```
    CoreToAgentMessage::SetAssist { assist_type, enabled } => {
        match assist_type.as_str() {
            "abs" => {
                // Read current state
                let current = adapter.read_assist_state().map(|(abs, _, _)| abs > 0).unwrap_or(false);
                if current != enabled {
                    if enabled {
                        // Turn ON: one Ctrl+A to go from 0 to level 1
                        ac_launcher::mid_session::toggle_ac_abs();
                    } else {
                        // Turn OFF: need to cycle down from current level to 0
                        let level = adapter.read_assist_state().map(|(abs, _, _)| abs).unwrap_or(1);
                        for _ in 0..level {
                            ac_launcher::mid_session::send_ctrl_shift_key(0x41); // Ctrl+Shift+A cycles down
                            tokio::time::sleep(Duration::from_millis(50)).await; // Small gap between keypresses
                        }
                    }
                }
                // Brief delay then confirm from shared memory
                tokio::time::sleep(Duration::from_millis(100)).await;
                let confirmed_abs = adapter.read_assist_state().map(|(abs, _, _)| abs > 0).unwrap_or(false);
                overlay.show_toast(if confirmed_abs { "ABS: ON".into() } else { "ABS: OFF".into() });
                let confirm = AgentMessage::AssistChanged {
                    pod_id: pod_id.clone(),
                    assist_type: "abs".into(),
                    enabled: confirmed_abs,
                    confirmed: true,
                };
                send_ws_message(&ws_tx, &confirm).await;
            }
            "tc" => {
                // Same pattern as ABS but with Ctrl+T / Ctrl+Shift+T
                let current = adapter.read_assist_state().map(|(_, tc, _)| tc > 0).unwrap_or(false);
                if current != enabled {
                    if enabled {
                        ac_launcher::mid_session::toggle_ac_tc();
                    } else {
                        let level = adapter.read_assist_state().map(|(_, tc, _)| tc).unwrap_or(1);
                        for _ in 0..level {
                            ac_launcher::mid_session::send_ctrl_shift_key(0x54);
                            tokio::time::sleep(Duration::from_millis(50)).await;
                        }
                    }
                }
                tokio::time::sleep(Duration::from_millis(100)).await;
                let confirmed_tc = adapter.read_assist_state().map(|(_, tc, _)| tc > 0).unwrap_or(false);
                overlay.show_toast(if confirmed_tc { "TC: ON".into() } else { "TC: OFF".into() });
                let confirm = AgentMessage::AssistChanged {
                    pod_id: pod_id.clone(),
                    assist_type: "tc".into(),
                    enabled: confirmed_tc,
                    confirmed: true,
                };
                send_ws_message(&ws_tx, &confirm).await;
            }
            "transmission" => {
                // Ctrl+G is a simple toggle, not a cycle
                let current = adapter.read_assist_state().map(|(_, _, auto)| auto).unwrap_or(false);
                if current != enabled {
                    ac_launcher::mid_session::toggle_ac_transmission();
                }
                tokio::time::sleep(Duration::from_millis(100)).await;
                let confirmed = adapter.read_assist_state().map(|(_, _, auto)| auto).unwrap_or(false);
                overlay.show_toast(if confirmed { "Transmission: AUTO".into() } else { "Transmission: MANUAL".into() });

                    pod_id: pod_id.clone(),
                    assist_type: "transmission".into(),
                    enabled: confirmed,
                    confirmed: true,
                };
                send_ws_message(&ws_tx, &confirm).await;
            }
            other => {
                tracing::warn!("Unknown assist type: {}", other);
            }
        }
    }
    ```
