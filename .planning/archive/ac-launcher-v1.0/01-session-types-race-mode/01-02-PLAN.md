```rust
    "ks_lamborghini_huracan_performante",
    "ks_porsche_911_gt3_cup_2017",
    "ks_mclaren_p1",
];

fn generate_trackday_ai(count: usize, player_car: &str) -> Vec<AiCarSlot> {
    use rand::seq::SliceRandom;
    let mut rng = rand::thread_rng();

    let names = pick_ai_names(count);
    let mut cars: Vec<&str> = TRACKDAY_CAR_POOL.to_vec();
    cars.shuffle(&mut rng);

    (0..count).map(|i| {
        AiCarSlot {
            model: cars[i % cars.len()].to_string(),
            skin: String::new(), // AC picks random
            driver_name: names[i].clone(),
            ai_level: 85, // Medium difficulty for casual track day
        }
    }).collect()
}
```

```rust
// Race Weekend: P -> Q -> R sequence
// Time allocation:
//   Practice: weekend_practice_minutes (0 = skip)
//   Qualifying: weekend_qualify_minutes (0 = skip)
//   Race: duration_minutes - practice - qualify (minimum 1 minute)

let mut session_index = 0;

if params.weekend_practice_minutes > 0 {
    // [SESSION_{session_index}] TYPE=1, NAME=Practice
    write_session_block(ini, session_index, "Practice", 1, params.weekend_practice_minutes, 1);
    session_index += 1;
}

if params.weekend_qualify_minutes > 0 {
    // [SESSION_{session_index}] TYPE=2, NAME=Qualifying
    write_session_block(ini, session_index, "Qualifying", 2, params.weekend_qualify_minutes, 1);
    session_index += 1;
}

// Race gets remaining time, minimum 1 minute
let race_time = params.duration_minutes
    .saturating_sub(params.weekend_practice_minutes)
    .saturating_sub(params.weekend_qualify_minutes)
    .max(1);
// [SESSION_{session_index}] TYPE=3, NAME=Race
write_session_block(ini, session_index, "Race", 3, race_time, params.starting_position);
```
