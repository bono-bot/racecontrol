```javascript
// #pending = new Map();  // Map<execId, { resolve, reject, timer }>

#waitForExecResult(execId, timeoutMs) {
  return new Promise((resolve, reject) => {
    const timer = setTimeout(() => {
      this.#pending.delete(execId);
      reject(new Error(`exec_result for ${execId} timed out after ${timeoutMs}ms`));
    }, timeoutMs);
    this.#pending.set(execId, { resolve, reject, timer });
  });
}

handleExecResult(payload) {
  const { execId } = payload ?? {};
  if (!execId) return;
  const pending = this.#pending.get(execId);
  if (!pending) return;
  clearTimeout(pending.timer);
  this.#pending.delete(execId);
  pending.resolve(payload);
}
```
