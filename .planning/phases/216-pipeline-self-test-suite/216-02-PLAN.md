Shebang, comment block identifying TEST-04, set -u; set -o pipefail.
Path setup + _pass/_fail helpers. Opening banner.

COORD-LOCK-WRITE:
  ( tmp_dir=$(mktemp -d)
    export REPO_ROOT COORD_LOCK_FILE="$tmp_dir/lock.json" RELAY_URL="http://localhost:8766"
    source "$REPO_ROOT/scripts/coordination/coord-state.sh" 2>/dev/null
    write_active_lock
    [[ -f "$COORD_LOCK_FILE" ]] || { rm -rf "$tmp_dir"; exit 1; }
    agent=$(jq -r .agent "$COORD_LOCK_FILE" 2>/dev/null || echo "")
    pid_val=$(jq -r .pid "$COORD_LOCK_FILE" 2>/dev/null || echo "0")
    rm -rf "$tmp_dir"
    [[ "$agent" == "james" ]] && [[ "$pid_val" -gt 0 ]] 2>/dev/null && exit 0 || exit 1
  ) 2>/dev/null

COORD-LOCK-CLEAR:
  ( tmp_dir=$(mktemp -d)
    export REPO_ROOT COORD_LOCK_FILE="$tmp_dir/lock.json" RELAY_URL="http://localhost:8766"
    source "$REPO_ROOT/scripts/coordination/coord-state.sh" 2>/dev/null
    write_active_lock; clear_active_lock
    [[ ! -f "$COORD_LOCK_FILE" ]] && exit 0 || exit 1
    rm -rf "$tmp_dir"
  ) 2>/dev/null

COORD-STALE-DETECT (fresh marker):
  ( tmp_dir=$(mktemp -d)
    export REPO_ROOT COORD_LOCK_FILE="$tmp_dir/lock.json" RELAY_URL="http://localhost:8766"
    export COORD_COMPLETION_FILE="$tmp_dir/completion.json"
    export COORD_STALE_SECS=600
    source "$REPO_ROOT/scripts/coordination/coord-state.sh" 2>/dev/null
    write_completion_marker "ok" 0 0
    is_james_run_recent && exit 0 || exit 1
    rm -rf "$tmp_dir"
  ) 2>/dev/null

COORD-STALE-EXPIRED (epoch 1000):
  ( tmp_dir=$(mktemp -d)
    export REPO_ROOT COORD_LOCK_FILE="$tmp_dir/lock.json" RELAY_URL="http://localhost:8766"
    export COORD_COMPLETION_FILE="$tmp_dir/completion.json"
    export COORD_STALE_SECS=600
    source "$REPO_ROOT/scripts/coordination/coord-state.sh" 2>/dev/null
    printf '{"agent":"james","completed_ts":1000,"verdict":"ok","bugs_found":0,"bugs_fixed":0,"run_dir":"x"}\n' > "$COORD_COMPLETION_FILE"
    is_james_run_recent && exit 1 || exit 0
    rm -rf "$tmp_dir"
  ) 2>/dev/null

COORD-MUTEX-RACE:
  ( tmp_dir=$(mktemp -d)
    export REPO_ROOT RELAY_URL="http://localhost:8766"
    lock_file="$tmp_dir/lock.json"
    source "$REPO_ROOT/scripts/coordination/coord-state.sh" 2>/dev/null
    export COORD_LOCK_FILE="$lock_file"
    ( export COORD_LOCK_FILE="$lock_file"; write_active_lock ) &
    pid1=$!
    ( export COORD_LOCK_FILE="$lock_file"; write_active_lock ) &
    pid2=$!
    wait "$pid1" "$pid2"
    [[ -f "$lock_file" ]] || { rm -rf "$tmp_dir"; exit 1; }
    agent=$(jq -r .agent "$lock_file" 2>/dev/null || echo "")
    rm -rf "$tmp_dir"
    [[ "$agent" == "james" ]] && exit 0 || exit 1
  ) 2>/dev/null

COORD-SYNTAX: bash -n on coord-state.sh.
