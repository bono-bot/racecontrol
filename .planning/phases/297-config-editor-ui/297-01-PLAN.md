GET /api/v1/config/pod/{pod_id}
Response (200): { pod_id: string, config: AgentConfig, config_hash: string, last_modified: string | null }

GET /api/v1/config/audit

  id: number,
  action: string,        // 'config_push' | 'full_config_set'

  entity_name: string,   // for full_config_set: pod_id; for config_push: comma-joined field names
  old_value: string | null,
  new_value: string | null,  // for full_config_set: config_hash
  pushed_by: string,     // staff sub from JWT (staff username)
  pods_acked: string,    // JSON array string e.g. '[]' or '["pod-1","pod-2"]'
  seq_num: number | null,
  created_at: string | null
}
