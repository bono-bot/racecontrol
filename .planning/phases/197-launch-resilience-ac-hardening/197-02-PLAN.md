```rust
/// Pre-launch health checks: verify pod is ready to launch a game.
/// Called synchronously in LaunchGame handler before spawning any game process.
/// Returns Ok(()) if all checks pass, Err(String) with specific reason if any fails.
pub fn pre_launch_checks() -> Result<(), String> {
    let rp_dir = std::path::Path::new("C:\\RacingPoint");

    // Check 1: No MAINTENANCE_MODE sentinel
    if rp_dir.join("MAINTENANCE_MODE").exists() {
        return Err("MAINTENANCE_MODE active — launch blocked".to_string());
    }

    // Check 2: No OTA_DEPLOYING sentinel
    if rp_dir.join("OTA_DEPLOYING").exists() {
        return Err("OTA_DEPLOYING active — launch blocked during OTA".to_string());
    }

    // Check 3: No orphan game processes running
    {
        use sysinfo::System;
        let mut sys = System::new();
        sys.refresh_processes(sysinfo::ProcessesToUpdate::All, true);
        let known = all_game_process_names();
        for (_pid, proc) in sys.processes() {
            let pname = proc.name().to_string_lossy().to_string();
            for name in known {
                if pname.eq_ignore_ascii_case(name) {
                    return Err(format!("orphan game process {} (PID {}) still running — clean state required", name, _pid.as_u32()));
                }
            }
        }
    }

    // Check 4: Disk space > 1GB (reuse sysinfo pattern from pre_flight.rs)
    {
        use sysinfo::Disks;
        let disks = Disks::new_with_refreshed_list();
        if let Some(d) = disks.iter().find(|d| {
            d.mount_point().to_str().map(|s| s == "C:\\" || s == "C:" || s == "/").unwrap_or(false)
        }) {
            if d.available_space() < 1_000_000_000 {
                return Err(format!("disk space low: {}MB free (< 1GB required)", d.available_space() / 1_048_576));
            }
        }
    }

    Ok(())
}
```

**2. Add clean_state_reset() function to game_process.rs:**

```rust
/// Full clean state reset: kill ALL game processes, clear game.pid, clear shared memory adapter file.
/// Called before auto-retry to ensure a clean slate.
/// Returns the number of processes killed.
pub fn clean_state_reset() -> u32 {
    let mut killed = 0u32;

    // Kill all known game processes
    use sysinfo::System;
    let mut sys = System::new();
    sys.refresh_processes(sysinfo::ProcessesToUpdate::All, true);
    let known = all_game_process_names();
    for (_pid, proc) in sys.processes() {
        let pname = proc.name().to_string_lossy().to_string();
        for name in known {
            if pname.eq_ignore_ascii_case(name) {
                let pid = _pid.as_u32();
                tracing::warn!(target: LOG_TARGET, pid, name, "clean_state_reset: killing game process");
                if kill_process(pid).is_ok() {
                    killed += 1;
                }
                break;
            }
        }
    }

    // Clear game.pid file
    clear_persisted_pid();

    // Clear shared memory adapter file if present
    let adapter_path = std::path::Path::new("C:\\RacingPoint\\shared_memory_adapter.lock");
    if adapter_path.exists() {
        let _ = std::fs::remove_file(adapter_path);
        tracing::info!(target: LOG_TARGET, "clean_state_reset: removed shared_memory_adapter.lock");
    }

    tracing::info!(target: LOG_TARGET, killed, "clean_state_reset complete");
    killed
}
```

```rust
fn wait_for_ac_ready(max_wait_secs: u64) -> bool {
    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(max_wait_secs);
    let mut pid_first_seen: Option<(u32, std::time::Instant)> = None;
    let mut last_log = std::time::Instant::now();

    while std::time::Instant::now() < deadline {
        match find_acs_pid() {
            Some(pid) => {
                let entry = pid_first_seen.get_or_insert((pid, std::time::Instant::now()));
                // Consider ready if same PID alive for 3+ seconds
                if entry.0 == pid && entry.1.elapsed().as_secs() >= 3 {
                    tracing::info!("AC process {} stable for 3s — ready", pid);
                    return true;
                }
                // PID changed (crash + respawn) — reset timer
                if entry.0 != pid {
                    *entry = (pid, std::time::Instant::now());
                }
            }
            None => {
                pid_first_seen = None;
            }
        }
        if last_log.elapsed() >= std::time::Duration::from_secs(5) {
            let elapsed = max_wait_secs.saturating_sub(
                deadline.duration_since(std::time::Instant::now()).as_secs()
            );
            tracing::info!("Waiting for AC process to stabilize... (~{}s elapsed)", elapsed);
            last_log = std::time::Instant::now();
        }
        std::thread::sleep(std::time::Duration::from_millis(500));
    }
    tracing::warn!("AC process not stable after {}s — proceeding", max_wait_secs);
    false
}
```

Use the simpler PID-stability approach. Call `wait_for_ac_ready(30);` where the old `sleep(8s)` was.

```rust
fn wait_for_ac_process(timeout_secs: u64) -> Result<u32> {
    let start = std::time::Instant::now();
    let deadline = start + std::time::Duration::from_secs(timeout_secs);
    let mut last_log = start;

    loop {
        if let Some(pid) = find_acs_pid() {
            tracing::info!("Found acs.exe (PID {}) after {:.1}s", pid, start.elapsed().as_secs_f64());
            return Ok(pid);
        }
        if std::time::Instant::now() >= deadline {
            anyhow::bail!("acs.exe not found after {}s timeout", timeout_secs);
        }
        // AC-03: Progress logging at 5s intervals
        if last_log.elapsed() >= std::time::Duration::from_secs(5) {
            tracing::info!("CM progress: checking acs.exe... ({:.0}s elapsed)", start.elapsed().as_secs_f64());
            last_log = std::time::Instant::now();
        }
        std::thread::sleep(std::time::Duration::from_millis(500));
    }
}
```
