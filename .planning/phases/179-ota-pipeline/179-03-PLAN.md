    5. **rollback_wave() — auto-rollback for health gate failure (OTA-04, OTA-07):**

       **CRITICAL: Rollback execution MUST use rc-sentry :8091/exec, NOT rc-agent :8090/exec.**
       The rollback bat kills rc-agent (taskkill /F /IM rc-agent.exe) — if executed via rc-agent :8090, the process serving the exec endpoint is killed mid-execution, and subsequent commands (move, start) may never run. rc-sentry (:8091) survives the kill because it is a separate binary. This is the exact banned pattern from CLAUDE.md standing rules: "NEVER use taskkill /F /IM rc-agent.exe followed by start in the same exec chain [via rc-agent]."

       ```rust
       pub async fn rollback_wave(
           state: &Arc<AppState>,
           pod_numbers: &[u32],
           manifest: &ReleaseManifest,
       ) -> Result<(), PipelineError> {
           tracing::warn!("OTA: Rolling back wave {:?}", pod_numbers);

           for pod_num in pod_numbers {
               let pod_id = format!("pod_{}", pod_num);
               let pod_ip = {
                   let pods = state.pods.read().await;
                   pods.get(&pod_id).map(|p| p.ip_address.clone())
               };
               let pod_ip = match pod_ip {
                   Some(ip) => ip,
                   None => continue,
               };

               // Write do-rollback.bat via rc-agent /write endpoint (rc-agent is still alive at this point)
               let write_url = format!("http://{}:8090/write", pod_ip);
               let write_result = state.http_client.post(&write_url)
                   .json(&serde_json::json!({
                       "path": r"C:\RacingPoint\do-rollback.bat",
                       "content": crate::deploy::ROLLBACK_SCRIPT_CONTENT,
                   }))
                   .timeout(Duration::from_secs(10))
                   .send()
                   .await;

               if write_result.is_err() {
                   tracing::error!("OTA: Failed to write rollback script to {}", pod_id);
                   continue;
               }

               // Execute rollback via rc-SENTRY :8091/exec (NOT rc-agent :8090/exec).
               // The bat kills rc-agent — executing via rc-agent would kill the exec handler.
               // rc-sentry is a separate process that survives the rc-agent kill.
               let exec_url = format!("http://{}:8091/exec", pod_ip);
               let _ = state.http_client.post(&exec_url)
                   .json(&serde_json::json!({
                       "cmd": r#"start /min cmd /c C:\RacingPoint\do-rollback.bat"#,
                       "timeout_ms": 5000,
                   }))
                   .timeout(Duration::from_secs(10))
                   .send()
                   .await;

               tracing::info!("OTA: Rollback triggered for {} via rc-sentry :8091", pod_id);
           }

           // Wait for rollback to complete (same pattern as deploy.rs ROLLBACK_VERIFY_DELAYS)
           tokio::time::sleep(Duration::from_secs(30)).await;

           Ok(())
       }
       ```

       check_sha256() {
           local pod_ip="$1"
           local expected_sha256="$2"

           if [ -z "$expected_sha256" ]; then
               echo "SKIP: No SHA256 provided for $pod_ip"
               return 0

           local result

           if [ "$actual_sha256" = "$expected_sha256" ]; then
               echo "PASS: SHA256 matches on $pod_ip"
               return 0
           else
               echo "FAIL: SHA256 mismatch on $pod_ip (got: $actual_sha256, expected: $expected_sha256)"
               return 1

       if [ -n "$MANIFEST_FILE" ]; then

           for pod_ip in "${POD_IPS[@]}"; do
