       ```rust
       pub async fn rollback_wave(

               let pod_ip = {

                   pods.get(&pod_id).map(|p| p.ip_address.clone())

                   }))
                   .timeout(Duration::from_secs(10))
                   .send()
                   .await;

               if write_result.is_err() {
                   tracing::error!("OTA: Failed to write rollback script to {}", pod_id);
                   continue;
               }

               // Execute rollback detached
               let exec_url = format!("http://{}:8090/exec", pod_ip);
               let _ = state.http_client.post(&exec_url)

                   .timeout(Duration::from_secs(10))
                   .send()
                   .await;

               tracing::info!("OTA: Rollback triggered for {}", pod_id);
           }

           // Wait for rollback to complete (same pattern as deploy.rs ROLLBACK_VERIFY_DELAYS)
           tokio::time::sleep(Duration::from_secs(30)).await;

           Ok(())
       }
       ```

       ```bash
       # SHA256 verification (OTA-10, SYNC-02)
       check_sha256() {
           local pod_ip="$1"
           local expected_sha256="$2"

           if [ -z "$expected_sha256" ]; then
               echo "SKIP: No SHA256 provided for $pod_ip"
               return 0
           fi

           # Compute SHA256 of rc-agent.exe on pod via certutil
           local result
           result=$(curl -s -X POST "http://$pod_ip:8090/exec" \
               -H "Content-Type: application/json" \
               -d "{\"cmd\":\"certutil -hashfile C:\\\\RacingPoint\\\\rc-agent.exe SHA256\",\"timeout_ms\":10000}" 2>/dev/null)

           local actual_sha256
           actual_sha256=$(echo "$result" | grep -o '"stdout":"[^"]*"' | sed 's/"stdout":"//;s/"//' | tr -d '\\n' | awk 'NR==2{print $1}')

           if [ "$actual_sha256" = "$expected_sha256" ]; then
               echo "PASS: SHA256 matches on $pod_ip"
               return 0
           else
               echo "FAIL: SHA256 mismatch on $pod_ip (got: $actual_sha256, expected: $expected_sha256)"
               return 1
           fi
       }

       # If --manifest flag provided, validate SHA256 against manifest
       if [ -n "$MANIFEST_FILE" ]; then
           EXPECTED_SHA=$(grep rc_agent_sha256 "$MANIFEST_FILE" | cut -d'"' -f2)
           for pod_ip in "${POD_IPS[@]}"; do
               check_sha256 "$pod_ip" "$EXPECTED_SHA"
           done
       fi
       ```

       Integrate this into the existing script's flow. The SHA256 check is optional (only when `--manifest` or `MANIFEST_FILE` is provided) so existing usage is not broken.
