       ```rust
       pub async fn run_pipeline(
           state: Arc<AppState>,
           manifest: ReleaseManifest,
       ) -> Result<(), PipelineError> {
           // Collect all pod numbers across all waves
           let all_pods: Vec<u32> = [WAVE_1, WAVE_2, WAVE_3].iter().flat_map(|w| w.iter().copied()).collect();

           // Set sentinels on ALL pods at pipeline start (OTA-09)
           set_ota_sentinel(&state, &all_pods).await;
           set_kill_switch(&state, true).await;

           let result = run_pipeline_inner(&state, &manifest).await;

           // ALWAYS clear sentinels — success or failure (OTA-09)
           clear_ota_sentinel(&state, &all_pods).await;
           set_kill_switch(&state, false).await;

           // Handle errors: rollback on HealthGateFailed
           match result {
               Ok(()) => Ok(()),
               Err(PipelineError::HealthGateFailed { wave, ref failures }) => {
                   let wave_pods = match wave {
                       1 => WAVE_1,
                       2 => WAVE_2,
                       3 => WAVE_3,
                       _ => &[],
                   };
                   // Update state to RollingBack
                   let mut record = match load_pipeline_state() {
                       Some(existing) => existing,
                       None => {
                           tracing::warn!("No pipeline state found during rollback — creating new record");
                           DeployRecord {
                               state: PipelineState::RollingBack,
                               manifest_version: manifest.release.version.clone(),
                               started_at: chrono::Utc::now().to_rfc3339(),
                               updated_at: chrono::Utc::now().to_rfc3339(),
                               waves_completed: wave.saturating_sub(1),
                               failed_pods: failures.iter().map(|f| f.pod_id.clone()).collect(),
                               rollback_reason: Some(format!("Health gate failed at wave {}", wave)),
                           }
                       }
                   };
                   record.state = PipelineState::RollingBack;
                   record.rollback_reason = Some(format!("Health gate failed at wave {}", wave));
                   if let Err(e) = persist_pipeline_state(&record) {
                       tracing::error!("Failed to persist rollback state: {}", e);
                   }

                   // Trigger rollback
                   if let Err(e) = rollback_wave(&state, wave_pods, &manifest).await {
                       tracing::error!("Rollback failed for wave {}: {:?}", wave, e);
                   }

                   result  // Return the original error
               }
               Err(e) => Err(e),
           }
       }
       ```

       The existing `run_pipeline` from Plan 02 becomes `run_pipeline_inner` (private), and the public `run_pipeline` wraps it with sentinel lifecycle and rollback handling.

    2. **Startup interrupt detection (phase goal: "impossible to interrupt without a trace"):**

       ```rust
       /// Called at racecontrol startup. If deploy-state.json has a non-terminal state,
       /// the server was restarted mid-deploy. Log a warning and mark the pipeline as interrupted.
       pub fn check_interrupted_pipeline() {
           match load_pipeline_state() {
               Some(mut record) => {
                   match record.state {
                       PipelineState::Idle | PipelineState::Completed => {
                           // Terminal states — no action needed
                       }
                       ref state => {
                           tracing::warn!(
                               pipeline_state = ?state,
                               manifest_version = %record.manifest_version,
                               waves_completed = record.waves_completed,
                               "OTA pipeline was interrupted by server restart — marking as interrupted"
                           );
                           record.state = PipelineState::RollingBack;
                           record.rollback_reason = Some("server_restart_interrupted".to_string());
                           record.updated_at = chrono::Utc::now().to_rfc3339();
                           if let Err(e) = persist_pipeline_state(&record) {
                               tracing::error!("Failed to persist interrupted pipeline state: {}", e);
                           }
                       }
                   }
               }
               None => {
                   // No state file — no pipeline was in progress
               }
           }
       }
       ```

       Add call to `check_interrupted_pipeline()` in `main.rs` during server startup, after tracing init but before server listen.

    3. **API endpoint: POST /api/v1/ota/deploy:**

       ```rust
       pub async fn ota_deploy_handler(
           State(state): State<Arc<AppState>>,
           body: String,
       ) -> impl axum::response::IntoResponse {
           // Parse manifest from request body (TOML string)
           let manifest = match parse_manifest(&body) {
               Ok(m) => m,
               Err(e) => return (StatusCode::BAD_REQUEST, Json(serde_json::json!({ "error": e }))),
           };

           // Check if a pipeline is already running
           if let Some(record) = load_pipeline_state() {
               if record.state != PipelineState::Idle && record.state != PipelineState::Completed {
                   return (StatusCode::CONFLICT, Json(serde_json::json!({
                       "error": "Pipeline already running",
                       "state": format!("{:?}", record.state),
                   })));
               }
           }

           // Spawn pipeline as background task (returns immediately)
           let state_clone = state.clone();
           tokio::spawn(async move {
               match run_pipeline(state_clone, manifest).await {
                   Ok(()) => tracing::info!("OTA pipeline completed successfully"),
                   Err(e) => tracing::error!(error = %e, "OTA pipeline failed"),
               }
           });

           (StatusCode::ACCEPTED, Json(serde_json::json!({ "status": "pipeline_started" })))
       }
       ```

       **Note on tokio::spawn:** The `.unwrap_or_else()` pattern is NOT used. The closure directly matches Ok/Err with explicit tracing. If `tokio::spawn` itself returns a JoinError (panic in spawned task), that is already logged by tokio's default panic handler. The critical path here is that `run_pipeline` errors are always logged via `tracing::error!`.

       ```rust
       pub async fn ota_status_handler() -> impl axum::response::IntoResponse {
           match load_pipeline_state() {
               Some(record) => {
                   // Use match instead of unwrap_or_default to handle serialization
                   match serde_json::to_value(&record) {
                       Ok(json) => (StatusCode::OK, Json(json)),
                       Err(e) => {
                           tracing::warn!("Failed to serialize pipeline state: {}", e);
                           (StatusCode::OK, Json(serde_json::json!({
                               "state": "error",
                               "message": format!("Serialization error: {}", e)
                           })))
                       }
                   }
               }
               None => (StatusCode::OK, Json(serde_json::json!({ "state": "idle", "message": "No pipeline state" }))),
           }
       }
       ```
