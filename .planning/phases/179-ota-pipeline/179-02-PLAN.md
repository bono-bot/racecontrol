       ```rust
       pub async fn run_pipeline(
           state: Arc<AppState>,
           manifest: ReleaseManifest,
       ) -> Result<(), PipelineError> {
           let started_at = chrono::Utc::now().to_rfc3339();

           // Create initial DeployRecord
           let mut record = DeployRecord {
               state: PipelineState::Canary,
               manifest_version: manifest.release.version.clone(),
               started_at: started_at.clone(),
               updated_at: started_at,
               waves_completed: 0,
               failed_pods: Vec::new(),
               rollback_reason: None,
           };

           let waves: &[&[u32]] = &[WAVE_1, WAVE_2, WAVE_3];

           for (wave_idx, wave) in waves.iter().enumerate() {
               let wave_num = (wave_idx + 1) as u8;

               // Update state based on wave index
               record.state = if wave_idx == 0 { PipelineState::Canary } else { PipelineState::StagedRollout };
               record.updated_at = chrono::Utc::now().to_rfc3339();
               persist_pipeline_state(&record).map_err(PipelineError::PersistFailed)?;

               // Deploy this wave
               deploy_wave(&state, wave, &manifest).await?;

               // Health gate
               record.state = PipelineState::HealthChecking;
               record.updated_at = chrono::Utc::now().to_rfc3339();
               persist_pipeline_state(&record).map_err(PipelineError::PersistFailed)?;

               // Retry with VERIFY_DELAYS pattern (5s, 15s, 30s, 60s)
               let mut last_failures = Vec::new();
               for delay in &[5u64, 15, 30, 60] {
                   tokio::time::sleep(Duration::from_secs(*delay)).await;
                   last_failures = check_wave_health(&state, wave, &manifest).await;
                   if last_failures.is_empty() { break; }
                   tracing::warn!("Health gate wave {}: {} pods failing, retrying", wave_num, last_failures.len());
               }

               if !last_failures.is_empty() {
                   // Health gate failed — trigger rollback for this wave
                   return Err(PipelineError::HealthGateFailed { wave: wave_num, failures: last_failures });
               }

               record.waves_completed = wave_num;
               tracing::info!("OTA pipeline: wave {} health gate PASSED", wave_num);
           }

           // All waves completed
           record.state = PipelineState::Completed;
           record.updated_at = chrono::Utc::now().to_rfc3339();
           persist_pipeline_state(&record).map_err(PipelineError::PersistFailed)?;

           Ok(())
       }
       ```

       pub async fn deploy_wave(
           state: &Arc<AppState>,
           pod_numbers: &[u32],
           manifest: &ReleaseManifest,
       ) -> Result<(), PipelineError> {
           let binary_url = format!("{}/rc-agent.exe", manifest.deploy.binary_url_base);

           for pod_num in pod_numbers {
               let pod_id = format!("pod_{}", pod_num);

               // Resolve pod IP
               let pod_ip = {
                   let pods = state.pods.read().await;
                   pods.get(&pod_id).map(|p| p.ip_address.clone())
               };
               let pod_ip = pod_ip.ok_or_else(|| PipelineError::PodNotFound(pod_id.clone()))?;

               // OTA-05: Check for active billing session
               let has_session = {
                   let timers = state.billing.active_timers.read().await;
                   timers.contains_key(&pod_id)
               };

               if has_session {
                   tracing::warn!("OTA: Pod {} has active billing session — waiting", pod_num);
                   // Wait up to 30 minutes for session to end
                   wait_for_session_end(state, &pod_id, Duration::from_secs(1800)).await?;
               }

               // Deploy this pod using existing deploy_pod()
               crate::deploy::deploy_pod(state.clone(), pod_id.clone(), pod_ip, binary_url.clone()).await;

               // Check deploy result
               let deploy_states = state.pod_deploy_states.read().await;
               if let Some(DeployState::Failed { reason }) = deploy_states.get(&pod_id) {
                   return Err(PipelineError::RollbackTriggered {
                       wave: 0, // caller knows the wave
                       reason: format!("Pod {} deploy failed: {}", pod_id, reason),
                   });
               }
           }
           Ok(())
       }
       ```

       async fn wait_for_session_end(
           state: &Arc<AppState>,
           pod_id: &str,
           timeout: Duration,
       ) -> Result<(), PipelineError> {
           let start = std::time::Instant::now();
           loop {
               if start.elapsed() > timeout {
                   return Err(PipelineError::SessionTimeout { pod_id: pod_id.to_string() });
               }
               let has_session = {
                   let timers = state.billing.active_timers.read().await;
                   timers.contains_key(pod_id)
               };
               if !has_session { return Ok(()); }
               tokio::time::sleep(Duration::from_secs(10)).await;
           }
       }
       ```
