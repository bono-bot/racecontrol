<objective>
Model split sessions as parent order + child entitlements with immutable durations, and enforce DB persistence before any split launch.

Purpose: A split session (e.g., 30min = 3x10min with different games) must have each sub-session tracked as an immutable child record. No launch can proceed without the split record being committed to DB first — preventing orphaned launches with no billing record.

Output: split_sessions table, SplitStatus FSM, split creation logic, DB-before-launch guard in game launcher.
</objective>

    /// FSM-07: Create child split entitlements for a parent session.
    /// Called when a session starts with split_count > 1.
    /// Each split gets an equal share of total allocated_seconds.
    pub async fn create_split_records(
        db: &sqlx::SqlitePool,
        parent_session_id: &str,
        split_count: u32,
        total_allocated_seconds: u32,
    ) -> Result<(), String> {
        let per_split = total_allocated_seconds / split_count;
        let remainder = total_allocated_seconds % split_count;
        for i in 1..=split_count {
            // Last split gets remainder seconds
            let alloc = if i == split_count { per_split + remainder } else { per_split };
            sqlx::query(
                "INSERT INTO split_sessions (parent_session_id, split_number, allocated_seconds, status) VALUES (?, ?, ?, 'pending')"
            )
            .bind(parent_session_id)
            .bind(i as i64)
            .bind(alloc as i64)
            .execute(db)
            .await
            .map_err(|e| format!("Failed to create split record {}: {}", i, e))?;
        }
        // Activate split 1 immediately
        sqlx::query(
            "UPDATE split_sessions SET status = 'active', started_at = datetime('now') WHERE parent_session_id = ? AND split_number = 1"
        )
        .bind(parent_session_id)
        .execute(db)
        .await
        .map_err(|e| format!("Failed to activate split 1: {}", e))?;
        Ok(())
    }

    /// FSM-07: Get the next pending split for a session (for split transitions).
    pub async fn get_next_pending_split(
        db: &sqlx::SqlitePool,
        parent_session_id: &str,
    ) -> Result<Option<(i64, i64)>, String> {
        // Returns (split_number, allocated_seconds)
        sqlx::query_as::<_, (i64, i64)>(
            "SELECT split_number, allocated_seconds FROM split_sessions WHERE parent_session_id = ? AND status = 'pending' ORDER BY split_number ASC LIMIT 1"
        )
        .bind(parent_session_id)
        .fetch_optional(db)
        .await
        .map_err(|e| format!("Failed to query next split: {}", e))
    }

    /// FSM-07: Complete current split and activate next.
    pub async fn transition_split(
        db: &sqlx::SqlitePool,
        parent_session_id: &str,
        current_split_number: i64,
    ) -> Result<Option<i64>, String> {
        // Complete current split (CAS: only if active)
        let completed = sqlx::query(
            "UPDATE split_sessions SET status = 'completed', ended_at = datetime('now') WHERE parent_session_id = ? AND split_number = ? AND status = 'active'"
        )
        .bind(parent_session_id)
        .bind(current_split_number)
        .execute(db)
        .await
        .map_err(|e| format!("Failed to complete split {}: {}", current_split_number, e))?;

        if completed.rows_affected() == 0 {
            return Err(format!("CAS failed: split {} not in active state", current_split_number));
        }

        // Activate next split
        let next = get_next_pending_split(db, parent_session_id).await?;
        if let Some((next_number, _)) = next {
            sqlx::query(
                "UPDATE split_sessions SET status = 'active', started_at = datetime('now') WHERE parent_session_id = ? AND split_number = ? AND status = 'pending'"
            )
            .bind(parent_session_id)
            .bind(next_number)
            .execute(db)
            .await
            .map_err(|e| format!("Failed to activate split {}: {}", next_number, e))?;
            Ok(Some(next_number))
        } else {
            Ok(None) // No more splits — session complete
        }
    }
    ```

    /// FSM-08: Transition to next split — persists to DB before issuing launch command.
    pub async fn transition_to_next_split(
        state: &Arc<AppState>,
        pod_id: &str,
        parent_session_id: &str,
        current_split: i64,
    ) -> Result<(), String> {
        // Step 1: Complete current split and activate next in DB
        let next_split = billing::transition_split(&state.db, parent_session_id, current_split).await?;

        match next_split {
            Some(next_number) => {
                // Step 2: Verify the split record exists in DB before launching
                let verified = sqlx::query_scalar::<_, i64>(
                    "SELECT COUNT(*) FROM split_sessions WHERE parent_session_id = ? AND split_number = ? AND status = 'active'"
                )
                .bind(parent_session_id)
                .bind(next_number)
                .fetch_one(&state.db)
                .await
                .map_err(|e| format!("FSM-08: DB verification failed: {}", e))?;

                if verified != 1 {
                    return Err(format!("FSM-08: Split {} not persisted as active in DB — aborting launch", next_number));
                }

                // Step 3: Update in-memory billing timer
                {
                    let mut timers = state.billing.active_timers.write().await;
                    if let Some(timer) = timers.get_mut(pod_id) {
                        timer.current_split_number = next_number as u32;
                        // Reset driving seconds for the new split
                        // (elapsed continues for parent session, but each split tracks its own time)
                    }
                }

                tracing::info!("FSM-08: Split {} persisted and verified — ready for launch on pod {}", next_number, pod_id);
                // The actual LaunchGame command will be issued by the caller after this returns Ok
                Ok(())
            }
            None => {
                // No more splits — session should end
                tracing::info!("FSM-08: All splits completed for session {} on pod {}", parent_session_id, pod_id);
                Err("All splits completed — session should end".to_string())
            }
        }
    }
    ```
