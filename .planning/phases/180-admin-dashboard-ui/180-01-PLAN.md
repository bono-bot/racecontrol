```typescript
const [status, setStatus] = useState<OtaStatusResponse | null>(null);
const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);

const pollStatus = useCallback(async () => {
  try {
    const res = await api.getOtaStatus();
    setStatus(res);
  } catch { /* ignore */ }
}, []);

useEffect(() => {
  pollStatus(); // immediate first fetch
  return () => { if (intervalRef.current) clearInterval(intervalRef.current); };
}, [pollStatus]);

useEffect(() => {
  if (intervalRef.current) clearInterval(intervalRef.current);
  const isActive = status && "waves_completed" in status && !["idle", "completed"].includes(status.state);
  const interval = isActive ? 3000 : 30000;
  intervalRef.current = setInterval(pollStatus, interval);
  return () => { if (intervalRef.current) clearInterval(intervalRef.current); };
}, [status, pollStatus]);
```
