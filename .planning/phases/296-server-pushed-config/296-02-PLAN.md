    **1. Add persist_server_config and load_server_config to config.rs:**

    ```rust
    /// PUSH-05: Persist server-pushed config to local JSON file for boot resilience.
    /// File: rc-agent-server-config.json in the same directory as the exe.
    pub fn persist_server_config(config: &AgentConfig, config_hash: &str) -> Result<()> {
        let path = server_config_path();
        let wrapper = serde_json::json!({
            "config": config,
            "config_hash": config_hash,
            "received_at": chrono::Utc::now().to_rfc3339(),
        });
        let json = serde_json::to_string_pretty(&wrapper)
            .map_err(|e| anyhow::anyhow!("Failed to serialize server config: {}", e))?;
        std::fs::write(&path, json)
            .map_err(|e| anyhow::anyhow!("Failed to write server config to {}: {}", path.display(), e))?;
        tracing::info!(target: "config", "Persisted server config to {} (hash={})", path.display(), config_hash);
        Ok(())
    }

    /// PUSH-05: Load last-received server config from local JSON file.
    /// Returns None if file doesn't exist or is corrupt.
    pub fn load_server_config() -> Option<(AgentConfig, String)> {
        let path = server_config_path();
        let content = std::fs::read_to_string(&path).ok()?;
        let wrapper: serde_json::Value = serde_json::from_str(&content).ok()?;
        let config: AgentConfig = serde_json::from_value(wrapper.get("config")?.clone()).ok()?;
        let hash = wrapper.get("config_hash")?.as_str()?.to_string();
        Some((config, hash))
    }

    fn server_config_path() -> std::path::PathBuf {
        std::env::current_exe()
            .ok()
            .and_then(|p| p.parent().map(|d| d.to_path_buf()))
            .unwrap_or_else(|| std::path::PathBuf::from("."))
            .join("rc-agent-server-config.json")
    }
    ```

    ```rust
    fn apply_full_config(state: &mut ConnState, new_config: &AgentConfig) {
        let old = &state.config;
        let mut pending_restart = Vec::new();

        // ─── Hot fields (apply immediately) ─────────────────────────────────
        // Process guard
        if old.process_guard.enabled != new_config.process_guard.enabled {
            tracing::info!(target: LOG_TARGET, "FullConfigPush: hot-reload process_guard.enabled = {}", new_config.process_guard.enabled);
            // Update via relevant state mechanism
        }
        if old.process_guard.scan_interval_secs != new_config.process_guard.scan_interval_secs {
            tracing::info!(target: LOG_TARGET, "FullConfigPush: hot-reload process_guard.scan_interval_secs = {}", new_config.process_guard.scan_interval_secs);
        }
        // ... repeat for other hot fields: kiosk.enabled, lock_screen.enabled, preflight.enabled,
        //     mma budgets, auto_end_orphan_session_secs, ac_evo_telemetry_enabled

        // ─── Cold fields (log pending-restart) ──────────────────────────────
        if old.pod.number != new_config.pod.number { pending_restart.push("pod.number"); }
        if old.core.url != new_config.core.url { pending_restart.push("core.url"); }
        if old.telemetry_ports.ports != new_config.telemetry_ports.ports { pending_restart.push("telemetry_ports.ports"); }
        if old.wheelbase.vendor_id != new_config.wheelbase.vendor_id { pending_restart.push("wheelbase.vendor_id"); }
        // ... other cold fields

        if !pending_restart.is_empty() {
            tracing::warn!(target: LOG_TARGET,
                "FullConfigPush: {} cold field(s) changed, require agent restart: {:?}",
                pending_restart.len(), pending_restart);
        }

        // Update state.config with the new config (so next hash comparison uses it)
        state.config = new_config.clone();
    }
    ```

    ```rust
    fn compute_config_hash_local(config: &AgentConfig) -> String {
        use sha2::{Sha256, Digest};
        let json = serde_json::to_string(config).unwrap_or_default();
        let hash = Sha256::digest(json.as_bytes());
        format!("{:x}", hash)
    }
    ```
