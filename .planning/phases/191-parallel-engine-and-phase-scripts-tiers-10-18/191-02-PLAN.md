Established phase script pattern (from tier1-9):
```bash
#!/usr/bin/env bash
# audit/phases/tierN/phaseNN.sh -- Phase NN: Title
# Tier: N (Category)
# What: Description

set -u
set -o pipefail
# NO set -e

run_phaseNN() {
  local phase="NN" tier="N"
  local mode="${AUDIT_MODE:-quick}"
  local venue_state="${VENUE_STATE:-unknown}"
  local response status severity message

  # ... checks using emit_result, http_get, safe_remote_exec ...

  # Venue-closed override for hardware/display checks:
  if [[ "$venue_state" = "closed" ]] && [[ "$status" = "FAIL" || "$status" = "WARN" ]]; then
    status="QUIET"; severity="P3"
  fi

  emit_result "$phase" "$tier" "$host" "$status" "$severity" "$message" "$mode" "$venue_state"
  return 0
}
export -f run_phaseNN
```

Available primitives (from core.sh):
- emit_result(phase tier host status severity message mode venue_state)

- get_session_token()
