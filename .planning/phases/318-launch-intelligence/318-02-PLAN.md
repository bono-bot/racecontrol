<objective>
Add step-level launch timeline tracing so every game launch records millisecond-resolution checkpoint spans (ws_sent, agent_received, process_spawned, playable_signal) that can be queried by launch_id to debug exactly where a launch stalled.

Purpose: When a customer can't launch a game, the support workflow is currently "check logs". Timeline spans make this "open dashboard, click launch_id, see checkpoint bars". Required by Phase 319 Reliability Dashboard.

Output: launch_timeline_spans SQLite table (server-side migration), LaunchTimelineReport WS send on agent, server WS handler that persists the timeline, GET /api/v1/launch-timeline/{launch_id} REST endpoint.
</objective>

```rust
// Helper function in ws_handler.rs:
fn build_launch_timeline(
    state: &crate::app_state::AppState,
    outcome: &str,
    extra_events: Vec<LaunchTimelineEvent>,
) -> Option<LaunchTimeline> {
    let launch_id = state.current_launch_id.as_ref()?.clone();
    let start = state.launch_start?;
    let elapsed = start.elapsed();
    let total_ms = elapsed.as_millis() as u64;
    let now_str = chrono::Utc::now().format("%Y-%m-%dT%H:%M:%S%.3fZ").to_string();

    let mut events = vec![
        LaunchTimelineEvent {
            kind: "agent_received".to_string(),
            detail: None,
            elapsed_ms: 0,
            timestamp: now_str.clone(),
        },
    ];
    events.extend(extra_events);
    // playable_signal = total_ms for success outcome
    if outcome == "success" {
        events.push(LaunchTimelineEvent {
            kind: "playable_signal".to_string(),
            detail: Some("BillingStarted received".to_string()),
            elapsed_ms: total_ms,
            timestamp: now_str.clone(),
        });
    }

    Some(LaunchTimeline {
        pod_id: state.pod_id.clone(),
        launch_id,
        billing_session_id: None, // filled in by caller if available
        sim_type: state.current_sim_type.unwrap_or(SimType::AssettoCorsa), // check app_state for current sim_type
        preset_id: None,
        events,
        outcome: outcome.to_string(),
        total_duration_ms: total_ms,
        started_at: now_str,
    })
}
```

```rust
async fn get_launch_timeline(
    State(state): State<Arc<AppState>>,
    Path(launch_id): Path<String>,
) -> Json<Value> {
    let row = sqlx::query_as::<_, (String, String, String, Option<String>, Option<String>, String, i64, String, String, String)>(
        "SELECT launch_id, pod_id, sim_type, preset_id, billing_session_id,
                outcome, total_duration_ms, started_at, events_json, created_at
         FROM launch_timeline_spans WHERE launch_id = ?"
    )
    .bind(&launch_id)
    .fetch_optional(&state.db)
    .await;

    match row {
        Ok(Some(r)) => {
            let events: Value = serde_json::from_str(&r.8).unwrap_or(Value::Array(vec![]));
            Json(json!({
                "launch_id": r.0,
                "pod_id": r.1,
                "sim_type": r.2,
                "preset_id": r.3,
                "billing_session_id": r.4,
                "outcome": r.5,
                "total_duration_ms": r.6,
                "started_at": r.7,
                "events": events,
                "created_at": r.9,
            }))
        }
        Ok(None) => Json(json!({ "launch_id": launch_id, "events": [] })),
        Err(e) => Json(json!({ "error": e.to_string() })),
    }
}
```
