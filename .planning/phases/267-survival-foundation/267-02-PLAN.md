**LeaseManager (in-memory, mutex-protected):**
```rust
use std::collections::HashMap;
use std::sync::Mutex;
use rc_common::survival_types::*;
use chrono::Utc;

pub struct LeaseManager {
    leases: Mutex<HashMap<String, HealLease>>,  // pod_id -> active lease
}

impl LeaseManager {
    pub fn new() -> Self { Self { leases: Mutex::new(HashMap::new()) } }

    pub fn request_lease(&self, req: &HealLeaseRequest) -> HealLeaseResponse {
        let mut leases = self.leases.lock().unwrap_or_else(|e| e.into_inner());
        // Check existing lease
        if let Some(existing) = leases.get(&req.pod_id) {
            // Check if expired
            let expires = chrono::DateTime::parse_from_rfc3339(&existing.expires_at)
                .map(|dt| dt < Utc::now())
                .unwrap_or(true);  // treat parse errors as expired
            if !expires {
                return HealLeaseResponse {
                    granted: false,
                    lease: None,
                    reason: Some(format!("pod {} already held by {:?} (action_id: {}, expires: {})",
                        req.pod_id, existing.granted_to, existing.action_id, existing.expires_at)),
                };
            }
            // Expired — remove and grant below
        }
        let now = Utc::now();
        let lease = HealLease {
            pod_id: req.pod_id.clone(),
            granted_to: req.layer,
            action_id: req.action_id.clone(),
            ttl_secs: req.ttl_secs,
            granted_at: now.to_rfc3339(),
            expires_at: (now + chrono::Duration::seconds(req.ttl_secs as i64)).to_rfc3339(),
        };
        tracing::info!(action_id = %req.action_id, pod_id = %req.pod_id,
            layer = ?req.layer, ttl_secs = req.ttl_secs,
            "heal lease GRANTED: {}", req.reason);
        leases.insert(req.pod_id.clone(), lease.clone());
        HealLeaseResponse { granted: true, lease: Some(lease), reason: None }
    }

    pub fn renew_lease(&self, pod_id: &str, layer: SurvivalLayer, action_id: &ActionId, ttl_secs: u64) -> Result<HealLease, String> {
        // Only the current holder can renew. Extend expires_at by ttl_secs from now.
    }

    pub fn release_lease(&self, pod_id: &str) {
        // Remove lease entry. Idempotent.
        tracing::info!(pod_id = pod_id, "heal lease RELEASED");
    }
}
```
