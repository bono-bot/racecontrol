   ```rust
   /// Write current flags to sentry-flags.json for rc-sentry consumption.
   /// Called after every FlagSync. rc-sentry reads this on its 5s watchdog cycle.
   pub fn write_sentry_flags(&self) {
       let path = std::path::Path::new(r"C:\RacingPoint\sentry-flags.json");
       let tmp_path = path.with_extension("json.tmp");
       let data = serde_json::json!({
           "flags": &self.flags,
           "kill_switches": &self.kill_switches,
           "version": self.version,
       });
       match serde_json::to_string_pretty(&data) {
           Ok(json) => {
               if let Err(e) = std::fs::write(&tmp_path, &json)
                   .and_then(|_| std::fs::rename(&tmp_path, path))
               {
                   tracing::warn!(target: LOG_TARGET, "Failed to write sentry-flags.json: {}", e);
               }
           }
           Err(e) => tracing::warn!(target: LOG_TARGET, "Failed to serialize sentry flags: {}", e),
       }
   }
   ```

2. **ws_handler.rs** — add ConfigPush match arm (after KillSwitch handler from Plan 01). The handler needs access to `ws_tx` to send ConfigAck back. Since `handle_ws_message` currently takes `state` and `conn`, and `ws_tx` is available in the event loop, the ConfigPush handler should:

   Define a list of hot-reloadable field names:
   ```rust
   const HOT_RELOAD_FIELDS: &[&str] = &["billing_rates", "game_limits", "process_guard_whitelist", "debug_verbosity"];
   const NON_RELOAD_FIELDS: &[&str] = &["port", "ws_url", "pod_number", "pod_id"];
   ```

   ```rust
   CoreToAgentMessage::ConfigPush(payload) => {
       let mut accepted = true;
       for (field, value) in &payload.fields {
           if NON_RELOAD_FIELDS.iter().any(|f| field.contains(f)) {
               tracing::warn!(target: LOG_TARGET, "ConfigPush: ignoring non-reloadable field '{}' (requires restart)", field);
               continue;
           }
           if field == "process_guard_whitelist" {
               // Update the existing guard_whitelist via its Arc<RwLock>
               if let Ok(wl) = serde_json::from_value::<MachineWhitelist>(value.clone()) {
                   let mut guard = state.guard_whitelist.write().await;
                   *guard = wl;
                   tracing::info!(target: LOG_TARGET, "ConfigPush: updated process_guard_whitelist");
               } else {
                   tracing::warn!(target: LOG_TARGET, "ConfigPush: invalid process_guard_whitelist value");
                   accepted = false;
               }
           } else {
               tracing::info!(target: LOG_TARGET, "ConfigPush: accepted field '{}' = {}", field, value);
               // Future hot-reload fields (billing_rates, game_limits, debug_verbosity)
               // will be wired here as their Arc<RwLock> state containers are created
           }
       }
       // Send ConfigAck back to server
       let ack = AgentMessage::ConfigAck(rc_common::types::ConfigAckPayload {
           pod_id: state.pod_id.clone(),
           sequence: payload.sequence,
           accepted,
       });
       conn.pending_acks.push(ack);
       tracing::info!(target: LOG_TARGET, "ConfigPush applied (seq={}), ack queued", payload.sequence);
   }
   ```

   For the ConfigAck send mechanism: `handle_ws_message` returns `HandleResult`. Add a `pending_acks: Vec<AgentMessage>` field to `ConnectionState` (in event_loop.rs). After `handle_ws_message` returns in the event loop, drain `pending_acks` and send each over `ws_tx`. This avoids passing `ws_tx` into handle_ws_message.

   Add `pub(crate) pending_acks: Vec<AgentMessage>,` to `ConnectionState` struct in event_loop.rs, initialize as `Vec::new()`. In the event loop where `handle_ws_message` result is processed, add:
   ```rust
   // Drain any acks queued by handlers
   for ack in conn.pending_acks.drain(..) {
       if let Ok(json) = serde_json::to_string(&ack) {
           let _ = ws_tx.send(Message::Text(json)).await;
       }
   }
   ```

   ```rust
   /// Read feature flags from sentry-flags.json (written by rc-agent on FlagSync).
   /// Returns None if file doesn't exist or can't be parsed.
   fn read_sentry_flags() -> Option<std::collections::HashMap<String, bool>> {
       let path = r"C:\RacingPoint\sentry-flags.json";
       let content = std::fs::read_to_string(path).ok()?;
       let parsed: serde_json::Value = serde_json::from_str(&content).ok()?;
       let flags = parsed.get("flags")?;
       serde_json::from_value(flags.clone()).ok()
   }
   ```
