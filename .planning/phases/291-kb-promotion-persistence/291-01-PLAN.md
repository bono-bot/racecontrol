```rust
impl KbPromotionStore {
    pub fn open(path: &str) -> anyhow::Result<Self>

    /// Insert or update a candidate. On conflict(problem_hash), update stage + stage_entered_at.

    /// Load all candidates at a given stage (e.g., "shadow", "canary").
    pub fn candidates_at_stage(&self, stage: &str) -> anyhow::Result<Vec<PromotionCandidate>>

    /// Load ALL candidates regardless of stage (used at startup to restore state).
    pub fn all_candidates(&self) -> anyhow::Result<Vec<PromotionCandidate>>

    pub fn update_stage(&self, problem_hash: &str, new_stage: &str) -> anyhow::Result<()>

    pub fn record_shadow_application(&self, problem_hash: &str) -> anyhow::Result<()>

    /// Get shadow_applications count for a given problem_hash.
    pub fn shadow_application_count(&self, problem_hash: &str) -> anyhow::Result<i64>
}
```

```rust
pub fn canary_gate(promo_store: &KbPromotionStore, problem_hash: &str, node_id: &str) -> bool {
    // Returns true if fix SHOULD be applied (i.e., pod IS canary pod OR stage != canary)
    // Returns false if fix should be SKIPPED (canary stage + not pod 8)
    let stage = promo_store.candidates_at_stage("canary")
        .ok()
        .and_then(|v| v.into_iter().find(|c| c.problem_hash == problem_hash))
        .map(|c| c.stage);

    match stage.as_deref() {
        Some("canary") => is_canary_pod(node_id),  // canary: only pod_8 applies
        _ => true,  // not canary stage: apply normally
    }
}
```
