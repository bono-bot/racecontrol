Lap time format helper (already in page.tsx):
```typescript
function formatLapTime(ms: number): string {
  if (ms <= 0) return "--:--.---";
  const totalSec = ms / 1000;
  const min = Math.floor(totalSec / 60);
  const sec = totalSec % 60;
  return `${min}:${sec.toFixed(3).padStart(6, "0")}`;
}
```

   ```typescript
   const wsRef = useRef<WebSocket | null>(null);
   useEffect(() => {
     const wsUrl = (process.env.NEXT_PUBLIC_WS_URL || "").replace("http", "ws");
     if (!wsUrl) return;
     const connect = () => {
       const ws = new WebSocket(wsUrl);
       wsRef.current = ws;
       ws.onmessage = (e) => {
         try {
           const msg = JSON.parse(e.data);
           if (msg.type === "LeaderboardUpdate" && Array.isArray(msg.records)) {
             setRecords(msg.records);
           }
         } catch {}
       };
       ws.onclose = () => setTimeout(connect, 2000); // 2s reconnect (min 1s per rules)
       ws.onerror = () => {};
     };
     connect();
     return () => { wsRef.current?.close(); };
   }, []);
   ```
