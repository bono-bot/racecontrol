```tsx
// Inline CountdownRing component (no imports needed — pure SVG)
function CountdownRing({ remaining, allocated }: { remaining: number; allocated: number }) {
  const CIRCUMFERENCE = 251.3; // 2 * PI * 40
  const pct = allocated > 0 ? Math.max(0, remaining / allocated) : 0;
  const dashOffset = CIRCUMFERENCE * (1 - pct);
  const isLow = remaining < 300;
  return (
    <div className="flex flex-col items-center gap-1">
      <svg width="64" height="64" className="-rotate-90">
        <circle cx="32" cy="32" r="26" fill="none" stroke="#333" strokeWidth="5" />
        <circle
          cx="32" cy="32" r="26" fill="none"
          stroke={isLow ? "#E10600" : "#22c55e"}
          strokeWidth="5"
          strokeDasharray={`${CIRCUMFERENCE * 26 / 40} ${CIRCUMFERENCE * 26 / 40}`}
          strokeDashoffset={(CIRCUMFERENCE * 26 / 40) * (1 - pct)}
          strokeLinecap="round"
          className={isLow ? "animate-pulse" : ""}
        />
      </svg>
      <span className={`text-xs font-mono ${isLow ? "text-rp-red animate-pulse" : "text-white"}`}>
        {Math.floor(remaining / 60)}:{String(Math.floor(remaining % 60)).padStart(2, "0")}
      </span>
    </div>
  );
}
```

Place the CountdownRing inside the pod card when an active session exists for that pod.
