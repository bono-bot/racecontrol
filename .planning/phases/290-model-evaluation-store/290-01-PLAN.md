**insert() method** — writes one EvalRecord row:
```rust
pub fn insert(&self, record: &EvalRecord) -> anyhow::Result<()> {
    self.conn.execute(
        "INSERT INTO model_evaluations \
         (id, model_id, pod_id, trigger_type, prediction, actual_outcome, correct, cost_usd, created_at) \
         VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8, ?9)",
        rusqlite::params![
            record.id, record.model_id, record.pod_id, record.trigger_type,
            record.prediction, record.actual_outcome,
            record.correct as i64, record.cost_usd, record.created_at
        ],
    )?;
    Ok(())
}
```

```rust
// EVAL-01: persist evaluation outcome to SQLite
{
    let trigger_name = format!("{:?}", &event.trigger)
        .split_whitespace()
        .next()
        .unwrap_or("Unknown")
        .to_string();
    let prediction_str: String = diagnosis.root_cause.chars().take(500).collect();
    let outcome_str = if fix_verified { "fixed" } else { "failed_to_fix" };
    let record = crate::model_eval_store::EvalRecord {
        id: uuid::Uuid::new_v4().to_string(),
        model_id: model_id.to_string(),
        pod_id: node_id.clone(),
        trigger_type: trigger_name,
        prediction: prediction_str,
        actual_outcome: outcome_str.to_string(),
        correct: fix_verified,
        cost_usd: step_cost,  // use the cost variable already tracked for this MMA step
        created_at: chrono::Utc::now().to_rfc3339(),
    };
    match eval_store.lock() {
        Ok(mut store) => {
            if let Err(e) = store.insert(&record) {
                tracing::warn!(
                    target: LOG_TARGET,
                    error = %e,
                    model = model_id,
                    "EVAL-01: failed to write evaluation record"
                );
            } else {
                tracing::debug!(
                    target: LOG_TARGET,
                    model = model_id,
                    correct = fix_verified,
                    "EVAL-01: evaluation record written"
                );
            }
        }
        Err(e) => {
            tracing::error!(target: LOG_TARGET, error = %e, "EVAL-01: eval_store mutex poisoned");
        }
    }
}
```

**Critical rule — never hold Mutex across .await**: The block above is synchronous (rusqlite insert). Ensure the entire `match eval_store.lock() { ... }` block completes before any `.await` point. The brace-enclosed block guarantees the MutexGuard is dropped before leaving the block.
