Create crates/racecontrol/src/game_inventory.rs with:

1. `pub async fn upsert_pod_game_inventory(db: &SqlitePool, inventory: &GameInventory) -> Result<(), sqlx::Error>`
   - For each InstalledGame in inventory.games: INSERT OR REPLACE into pod_game_inventory
   - sim_type stored as the Display string (e.g. "AssettoCorsa") via format!("{:?}", sim_type)
   - No .unwrap(); use ? throughout
   - LOG: tracing::info!(pod_id, game_count, "Persisted game inventory")

2. `pub async fn upsert_combo_validation(db: &SqlitePool, pod_id: &str, result: &ComboValidationResult) -> Result<(), sqlx::Error>`
   - INSERT OR REPLACE into combo_validation_flags
   - checked_paths stored as JSON array: serde_json::to_string(&result.checked_paths).unwrap_or_default()
   - status stored as Debug string: format!("{:?}", result.status)
   - No .unwrap() except on the serde_json fallback

3. `pub enum FleetValidity { Valid, Partial, Invalid }`
   `pub async fn compute_fleet_validity(db: &SqlitePool, preset_id: &str) -> Result<FleetValidity, sqlx::Error>`
   - Query: SELECT COUNT(*) as total, SUM(CASE WHEN status = 'Available' THEN 1 ELSE 0 END) as available FROM combo_validation_flags WHERE preset_id = ?
   - total = 0 → Invalid (no pod has validated this preset yet)
   - available == total && total > 0 → Valid
   - available > 0 && available < total → Partial
   - available == 0 && total > 0 → Invalid

4. `pub async fn auto_disable_invalid_combos(db: &SqlitePool, state: &AppState) -> Result<Vec<(String, String, String)>, sqlx::Error>`
   - Query presets that: (a) are currently enabled, (b) have at least one combo_validation_flags row, (c) have zero Available rows
   - SQL: SELECT gp.id, gp.name FROM game_presets gp WHERE gp.enabled = 1 AND (SELECT COUNT(*) FROM combo_validation_flags WHERE preset_id = gp.id) > 0 AND (SELECT COUNT(*) FROM combo_validation_flags WHERE preset_id = gp.id AND status = 'Available') = 0
   - For each: UPDATE game_presets SET enabled = 0 WHERE id = ?
   - Collect (preset_id, preset_name, "no pod has this combo available") tuples
   - Return collected tuples for caller to send WhatsApp alert

In ws/mod.rs, add two match arms in the AgentMessage handler block (after the EscalationRequest arm, before the ExperienceScoreReport arm):

```rust
// ─── Phase 317: Game inventory persistence ────────────────────────────────
AgentMessage::GameInventoryUpdate(inventory) => {
    let db = state.db.clone();
    let inv = inventory.clone();
    tokio::spawn(async move {
        if let Err(e) = crate::game_inventory::upsert_pod_game_inventory(&db, &inv).await {
            tracing::warn!(
                target: "racecontrol::ws",
                pod_id = %inv.pod_id,
                error = %e,
                "Failed to persist game inventory"
            );
        }
    });
}

// ─── Phase 317: Combo validation persistence + auto-disable ──────────────
AgentMessage::ComboValidationReport { pod_id, results } => {
    let db = state.db.clone();
    let pod_id_owned = pod_id.clone();
    let results_owned = results.clone();
    let state_clone = state.clone();
    tokio::spawn(async move {
        for result in &results_owned {
            if let Err(e) = crate::game_inventory::upsert_combo_validation(&db, &pod_id_owned, result).await {
                tracing::warn!(
                    target: "racecontrol::ws",
                    pod_id = %pod_id_owned,
                    preset_id = %result.preset_id,
                    error = %e,
                    "Failed to persist combo validation result"
                );
            }
        }
        // After all results persisted, check for newly-all-invalid combos
        match crate::game_inventory::auto_disable_invalid_combos(&db, &state_clone).await {
            Ok(disabled) => {
                for (preset_id, preset_name, reason) in disabled {
                    tracing::warn!(
                        target: "racecontrol::ws",
                        preset_id = %preset_id,
                        preset_name = %preset_name,
                        "Combo auto-disabled: not available on any pod"
                    );
                    // WhatsApp via EscalationRequest path (standing rule: never direct Evolution API)
                    let escalation = state_clone.whatsapp_escalation.clone();
                    let incident_id = format!("combo_disabled_{}", preset_id);
                    let msg = format!("Combo disabled: {} — not available on any pod. Missing: {}", preset_name, reason);
                    tokio::spawn(async move {
                        escalation.handle_escalation(rc_common::protocol::EscalationPayload {
                            pod_id: "server".to_string(),
                            incident_id,
                            severity: "high".to_string(),
                            trigger: "ComboAllPodsInvalid".to_string(),
                            summary: msg,
                            actions_tried: vec!["auto_disabled_preset".to_string()],
                            impact: "Preset disabled — customers cannot book this combo".to_string(),
                            dashboard_url: "http://192.168.31.23:3201/presets".to_string(),
                            timestamp: crate::whatsapp_alerter::ist_now_string(),
                        }).await;
                    });
                }
            }
            Err(e) => {
                tracing::warn!(target: "racecontrol::ws", error = %e, "auto_disable_invalid_combos failed");
            }
        }
    });
}
```
