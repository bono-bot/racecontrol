```rust
//! Watches C:\RacingPoint\ for sentinel file changes using notify 8.2.0.
//! Emits AgentMessage::SentinelChange over the agent's WS sender channel.

use notify::{RecommendedWatcher, RecursiveMode, Watcher, Event, EventKind};
use std::path::Path;
use tokio::sync::mpsc;
use rc_common::protocol::AgentMessage;

const WATCH_DIR: &str = r"C:\RacingPoint";
const KNOWN_SENTINELS: &[&str] = &[
    "MAINTENANCE_MODE",
    "GRACEFUL_RELAUNCH",
    "OTA_DEPLOYING",
    "rcagent-restart-sentinel.txt",
];

const LOG_TARGET: &str = "sentinel_watcher";

/// Spawn a background task that watches WATCH_DIR for sentinel file changes.
/// Sends AgentMessage::SentinelChange to the provided tx channel.
pub fn spawn(tx: mpsc::Sender<AgentMessage>, pod_id: String) {
    std::thread::spawn(move || {
        tracing::info!(target: LOG_TARGET, dir = WATCH_DIR, "sentinel watcher started");

        let (notify_tx, notify_rx) = std::sync::mpsc::channel();

        let mut watcher = match RecommendedWatcher::new(
            move |res: Result<Event, notify::Error>| {
                if let Ok(event) = res {
                    let _ = notify_tx.send(event);
                }
            },
            notify::Config::default(),
        ) {
            Ok(w) => w,
            Err(e) => {
                tracing::error!(target: LOG_TARGET, error = %e, "failed to create file watcher");
                return;
            }
        };

        if let Err(e) = watcher.watch(Path::new(WATCH_DIR), RecursiveMode::NonRecursive) {
            tracing::error!(target: LOG_TARGET, error = %e, "failed to watch directory");
            return;
        }

        loop {
            match notify_rx.recv() {
                Ok(event) => {
                    let action = match event.kind {
                        EventKind::Create(_) => "created",
                        EventKind::Remove(_) => "deleted",
                        _ => continue, // ignore modify, access, etc.
                    };

                    for path in &event.paths {
                        let file_name = match path.file_name().and_then(|n| n.to_str()) {
                            Some(name) => name,
                            None => continue,
                        };

                        if !KNOWN_SENTINELS.contains(&file_name) {
                            continue;
                        }

                        tracing::warn!(
                            target: "state",
                            sentinel = file_name,
                            action = action,
                            "sentinel file change detected"
                        );

                        // Timestamp in IST
                        let timestamp = chrono::Utc::now()
                            .with_timezone(&chrono::FixedOffset::east_opt(5 * 3600 + 30 * 60).unwrap())
                            .to_rfc3339();

                        let msg = AgentMessage::SentinelChange {
                            pod_id: pod_id.clone(),
                            file: file_name.to_string(),
                            action: action.to_string(),
                            timestamp: timestamp,
                        };

                        if tx.blocking_send(msg).is_err() {
                            tracing::error!(target: LOG_TARGET, "WS sender channel closed — stopping sentinel watcher");
                            return;
                        }
                    }
                }
                Err(_) => {
                    tracing::error!(target: LOG_TARGET, "notify channel closed — stopping sentinel watcher");
                    return;
                }
            }
        }
    });
}
```

use std::sync::Mutex;
use std::collections::HashMap;
use std::time::Instant;
use once_cell::sync::Lazy;

static SENTINEL_ALERT_COOLDOWN: Lazy<Mutex<HashMap<String, Instant>>> = Lazy::new(|| Mutex::new(HashMap::new()));

fn check_sentinel_cooldown(key: &str, cooldown_secs: u64) -> bool {
    let mut map = SENTINEL_ALERT_COOLDOWN.lock().unwrap_or_else(|p| p.into_inner());
    let now = Instant::now();
    if let Some(last) = map.get(key) {
        if now.duration_since(*last).as_secs() < cooldown_secs {
            return false;
        }
    }
    map.insert(key.to_string(), now);
    true
}
