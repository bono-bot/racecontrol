```rust
//! rc-process-guard — Standalone process guard for James workstation (.27).
//!
//! Reports via HTTP POST to racecontrol — never via WebSocket (standing rule #2).
//! Reads config from C:\Users\bono\racingpoint\rc-process-guard.toml.

use std::collections::HashMap;
use std::time::Duration;
use anyhow::Result;
use chrono::Utc;
use serde::Deserialize;
use rc_common::types::{MachineWhitelist, ProcessViolation, ViolationType};

const LOG_FILE: &str = r"C:\Users\bono\racingpoint\process-guard-james.log";
const MAX_LOG_BYTES: u64 = 512 * 1024;
const JAMES_CRITICAL_BINARIES: &[&str] = &["rc-agent.exe", "kiosk.exe"];

#[derive(Debug, Deserialize)]
struct GuardConfig {
    server_url: String,
    #[serde(default)]
    tailscale_url: Option<String>,
    #[serde(default)]
    report_secret: Option<String>,
    #[serde(default = "default_scan_interval")]
    scan_interval_secs: u64,
    #[serde(default = "default_machine_id")]
    machine_id: String,
}

fn default_scan_interval() -> u64 { 60 }
fn default_machine_id() -> String { "james".to_string() }

#[tokio::main]
async fn main() -> Result<()> {
    // Init tracing
    tracing_subscriber::fmt().with_max_level(tracing::Level::INFO).init();

    // Load config from C:\Users\bono\racingpoint\rc-process-guard.toml
    let config_path = r"C:\Users\bono\racingpoint\rc-process-guard.toml";
    let config: GuardConfig = {
        let raw = std::fs::read_to_string(config_path)
            .unwrap_or_else(|_| {
                tracing::warn!("Config not found at {}; using defaults", config_path);
                format!("server_url = \"http://192.168.31.23:8080\"")
            });
        toml::from_str(&raw)?
    };

    let client = reqwest::Client::builder()
        .timeout(Duration::from_secs(10))
        .build()?;

    // Fetch whitelist on startup (retry up to 5 times with 5s backoff)
    let whitelist = fetch_whitelist_with_retry(&client, &config).await;
    let whitelist = std::sync::Arc::new(tokio::sync::RwLock::new(whitelist));

    tracing::info!("rc-process-guard started (machine={}, interval={}s)", config.machine_id, config.scan_interval_secs);

    // 60s startup amnesty — mirrors rc-agent pattern
    tokio::time::sleep(Duration::from_secs(60)).await;

    let mut scan_interval = tokio::time::interval(Duration::from_secs(config.scan_interval_secs));
    let mut audit_interval = tokio::time::interval(Duration::from_secs(300));
    // grace_counts: process_name -> (consecutive_count, first_start_time)
    let mut grace_counts: HashMap<String, (u32, u64)> = HashMap::new();

    // Whitelist refresh interval (every 5 min alongside audit)
    let mut whitelist_refresh = tokio::time::interval(Duration::from_secs(300));

    loop {
        tokio::select! {
            _ = scan_interval.tick() => {
                run_scan_cycle(&whitelist, &client, &config, &mut grace_counts).await;
            }
            _ = audit_interval.tick() => {
                run_autostart_audit_james(&whitelist, &client, &config).await;
                run_port_audit_james(&whitelist, &client, &config).await;
            }
            _ = whitelist_refresh.tick() => {
                let fresh = fetch_whitelist_with_retry(&client, &config).await;
                *whitelist.write().await = fresh;
                tracing::info!("Whitelist refreshed");
            }
        }
    }
}
```

Implement these functions in the same file:

**fetch_whitelist_with_retry**: GET {server_url}/api/v1/guard/whitelist/james with up to 5 retries (5s sleep between). If Tailscale URL configured, try tailscale first. Returns MachineWhitelist::default() on total failure (report_only mode — safe).

**run_scan_cycle**: Mirror of rc-agent run_scan_cycle but:
- Self-exclude: own PID + "rc-process-guard.exe" name
- CRITICAL binaries: JAMES_CRITICAL_BINARIES (rc-agent.exe, kiosk.exe)
- Kill via tokio::task::spawn_blocking: taskkill /F /PID {pid} with #[cfg(windows)] creation_flags(0x08000000)
- Send violations via post_violation() (HTTP POST) instead of mpsc channel
- grace_counts same two-cycle logic as rc-agent

**post_violation**: async fn using reqwest::Client to POST {server_url}/api/v1/guard/report with JSON body (ProcessViolation). Sets header X-Guard-Token if config.report_secret is Some. On 401, logs warning. On connection error, logs warning + appends to local log (do not panic).

**log_james_event**: Same 512KB rotation pattern as log_guard_event in rc-agent but writes to LOG_FILE constant.

**run_autostart_audit_james**: Same logic as rc-agent run_autostart_audit — HKCU Run, HKLM Run, Startup folders, schtasks. Use Command::new("reg"), Command::new("schtasks") shell-outs (same as rc-agent). Violations sent via post_violation(). Reuse is_autostart_whitelisted logic inline.

**run_port_audit_james**: Same as Task 1 Plan 01 run_port_audit — netstat -ano parse, port whitelist check, kill PID, send via post_violation().

- `is_james_critical(name: &str) -> bool` — checks JAMES_CRITICAL_BINARIES
