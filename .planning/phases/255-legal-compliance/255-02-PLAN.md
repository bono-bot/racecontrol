    let is_minor = if let Some(ref dob_str) = dob {
        // Parse DOB and check if under 18
        if let Ok(dob_date) = chrono::NaiveDate::parse_from_str(dob_str, "%Y-%m-%d") {
            let today = chrono::Utc::now().date_naive();
            let age = today.years_since(dob_date).unwrap_or(99);
            age < 18
        } else {
            false // Can't determine age — treat as adult
        }
    } else {
        false // No DOB — treat as adult
    };
