Then add API call functions (follow existing pattern):
```typescript
export const listCafePromos = () =>
  api.get<CafePromo[]>("/cafe/promos");

export const createCafePromo = (data: CreateCafePromoRequest) =>
  api.post<CafePromo>("/cafe/promos", data);

export const updateCafePromo = (id: string, data: UpdateCafePromoRequest) =>
  api.put<CafePromo>(`/cafe/promos/${id}`, data);

export const deleteCafePromo = (id: string) =>
  api.delete<void>(`/cafe/promos/${id}`);

export const toggleCafePromo = (id: string) =>
  api.post<CafePromo>(`/cafe/promos/${id}/toggle`, {});
```
