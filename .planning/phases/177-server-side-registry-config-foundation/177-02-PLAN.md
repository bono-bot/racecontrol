**`validate_config_push(fields: &HashMap<String, serde_json::Value>) -> Result<(), HashMap<String, String>>`**

Whitelist validation:
- `"billing_rate"` — must be a number > 0. Error: `"must be a positive number"`
- `"game_limit"` — must be integer 1-10. Error: `"must be an integer between 1 and 10"`
- `"debug_verbosity"` — must be one of `["off", "error", "warn", "info", "debug", "trace"]`. Error: `"must be one of [off, error, warn, info, debug, trace]"`
- `"process_guard_whitelist"` — must be a non-empty JSON array of strings. Error: `"must be a non-empty array of strings"`
- Any unknown key: error `"unknown config field"`

Return `Ok(())` if no errors, `Err(errors_map)` otherwise.

**`push_config` handler**: `POST /api/v1/config/push`
- Extract `Extension(claims)`, `Json(body): Json<PushConfigRequest>`
- Call `validate_config_push(&body.fields)`. On Err, return `(StatusCode::BAD_REQUEST, Json(json!({"errors": field_errors})))`.
- Determine target pods: if `body.target_pods` is empty, get all pod IDs from `state.pods.read().await`. Otherwise use `body.target_pods`.
- Get a SINGLE seq_num for this push batch: `state.config_push_seq.fetch_add(1, Ordering::SeqCst)`. All pods in this batch share the same seq_num (they are the same config push event).
- For each target pod:
  1. Insert into `config_push_queue`: pod_id, payload (JSON string of fields), seq_num, status='pending'
  2. If pod is in `state.agent_senders`, send `CoreToAgentMessage::ConfigPush(ConfigPushPayload { fields: body.fields.clone(), schema_version: body.schema_version, sequence: seq_num })` via the sender. On success, update status to 'delivered'.
  3. If pod is NOT in agent_senders (offline), leave status='pending' (CP-02: offline queue)
- Insert audit log entry: action="config_push", entity_type="config", entity_name=comma-joined field keys, old_value=null, new_value=JSON of fields, pushed_by=claims.sub, pods_acked="[]", **seq_num=seq_num** (the same sequence number used in the queue entries)
- Return 200 with `{"queued": N, "delivered": M, "seq_nums": [seq_num]}`

**`get_queue` handler**: `GET /api/v1/config/push/queue`
- Query `SELECT * FROM config_push_queue ORDER BY id DESC LIMIT 100`
- Return `Json(Vec<ConfigQueueEntry>)`

**FlagCacheSync handler:**
```rust
AgentMessage::FlagCacheSync(payload) => {
    tracing::info!("Pod {} requests flag sync (cached_version={})", payload.pod_id, payload.cached_version);
    // Get current max flag version from state.feature_flags cache
    let max_version = {
        let flags = state.feature_flags.read().await;
        flags.values().map(|f| f.version).max().unwrap_or(0)
    };
    // If pod is stale, send full flag state
    if payload.cached_version < max_version as u64 {
        let flags = state.feature_flags.read().await;
        let flag_map: std::collections::HashMap<String, bool> = flags.iter()
            .map(|(name, row)| (name.clone(), row.enabled))
            .collect();
        let sync_payload = rc_common::types::FlagSyncPayload {
            flags: flag_map,
            version: max_version as u64,
        };
        let _ = cmd_tx.send(CoreToAgentMessage::FlagSync(sync_payload)).await;
    }
    // Replay pending config pushes for this pod.
    // IMPORTANT: Do NOT pass payload.cached_version as a sequence filter.
    // cached_version is a FLAG version counter — it has nothing to do with config push sequences.
    // replay_pending_config_pushes uses status != 'acked' as the filter instead.
    crate::config_push::replay_pending_config_pushes(
        &state, &payload.pod_id, &cmd_tx
    ).await;
}
```

**ConfigAck handler:**
```rust
AgentMessage::ConfigAck(payload) => {
    tracing::info!("Pod {} acked config push seq={} accepted={}", payload.pod_id, payload.sequence, payload.accepted);
    // Update config_push_queue: set status='acked', acked_at=datetime('now') WHERE pod_id=? AND seq_num=?
    let ack_result = sqlx::query(
        "UPDATE config_push_queue SET status = 'acked', acked_at = datetime('now') WHERE pod_id = ? AND seq_num = ?"
    )
    .bind(&payload.pod_id)
    .bind(payload.sequence as i64)
    .execute(&state.db)
    .await;
    if let Err(e) = ack_result {
        tracing::warn!("Failed to update config_push_queue ack for pod {} seq {}: {}", payload.pod_id, payload.sequence, e);
    }
    // Update config_audit_log pods_acked — find the audit entry by seq_num (deterministic lookup).
    // IMPORTANT: Do NOT use ORDER BY id DESC LIMIT 1 — that is non-deterministic under concurrent pushes.
    // The seq_num column was added to config_audit_log specifically for this lookup.
    let audit_update = sqlx::query_scalar::<_, i64>(
        "SELECT id FROM config_audit_log WHERE entity_type = 'config' AND seq_num = ?"
    )
    .bind(payload.sequence as i64)
    .fetch_optional(&state.db)
    .await;
    if let Ok(Some(audit_id)) = audit_update {
        let current_acked: String = sqlx::query_scalar(
            "SELECT pods_acked FROM config_audit_log WHERE id = ?"
        )
        .bind(audit_id)
        .fetch_one(&state.db)
        .await
        .unwrap_or_else(|_| "[]".to_string());
        let mut acked: Vec<String> = serde_json::from_str(&current_acked).unwrap_or_default();
        if !acked.contains(&payload.pod_id) {
            acked.push(payload.pod_id.clone());
        }
        let _ = sqlx::query(
            "UPDATE config_audit_log SET pods_acked = ? WHERE id = ?"
        )
        .bind(serde_json::to_string(&acked).unwrap_or_else(|_| "[]".to_string()))
        .bind(audit_id)
        .execute(&state.db)
        .await;
    } else {
        tracing::warn!("No audit log entry found for config push seq={}", payload.sequence);
    }
}
```
