    ```typescript
    export function getGameMatrix(): Promise<GameMatrixResponse> {
      return metricsGet<GameMatrixResponse>("/fleet/game-matrix");
    }

    export function getComboList(params?: { game?: string; sort_by?: string; order?: string }): Promise<ComboListRow[]> {
      const qs = params ? new URLSearchParams(
        Object.fromEntries(Object.entries(params).filter(([, v]) => v !== undefined && v !== "")) as Record<string, string>

      return metricsGet<ComboListRow[]>(`/admin/combo-list${qs ? `?${qs}` : ""}`);
    }
    ```
