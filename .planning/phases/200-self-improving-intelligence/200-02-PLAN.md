       AND total_launches >= 5
       AND NOT (COALESCE(car, '') = COALESCE(?, '') AND COALESCE(track, '') = COALESCE(?, ''))
     ORDER BY
       (CASE WHEN car = ? OR track = ? THEN 1 ELSE 0 END) DESC,
       success_rate DESC
     LIMIT 3
