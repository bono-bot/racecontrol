   ```rust
   use hmac::{Hmac, Mac};
   use sha2::Sha256;
   type HmacSha256 = Hmac<Sha256>;

   fn sign_sync_request(body: &[u8], key: &[u8]) -> (String, i64, String) {
       let timestamp = chrono::Utc::now().timestamp();
       let nonce = uuid::Uuid::new_v4().to_string();
       let mut mac = HmacSha256::new_from_slice(key).expect("HMAC accepts any key length");
       mac.update(&timestamp.to_be_bytes());
       mac.update(nonce.as_bytes());
       mac.update(body);
       let signature = hex::encode(mac.finalize().into_bytes());
       (signature, timestamp, nonce)
   }

   pub(crate) fn verify_sync_signature(
       body: &[u8], key: &[u8],
       timestamp: i64, nonce: &str, signature: &str,
   ) -> bool {
       let now = chrono::Utc::now().timestamp();
       if (now - timestamp).abs() > 300 { return false; }
       let mut mac = HmacSha256::new_from_slice(key).expect("HMAC accepts any key length");
       mac.update(&timestamp.to_be_bytes());
       mac.update(nonce.as_bytes());
       mac.update(body);
       mac.verify_slice(&hex::decode(signature).unwrap_or_default()).is_ok()
   }
   ```
