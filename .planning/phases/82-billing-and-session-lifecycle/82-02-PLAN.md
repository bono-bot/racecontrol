**Per-sim PlayableSignal dispatch in game_check_interval:**

4. In the `game_check_interval` tick (where the agent checks if a game process is running), add per-sim dispatch AFTER process detection:

```rust
// When process is first detected and launch_state is WaitingForLive:
// a) Emit GameState::Loading to server (once)
if !conn.loading_emitted && matches!(conn.launch_state, LaunchState::WaitingForLive { .. }) {
    // Send game_state_update with Loading to server
    // (existing DashboardEvent::GameStateChanged or direct WS message)
    conn.loading_emitted = true;
}

// b) Per-sim PlayableSignal check
match conn.current_sim_type {
    Some(SimType::AssettoCorsa) => {
        // Unchanged: AcStatus::Live from shared memory triggers billing
        // (handled in existing telemetry_interval arm)
    }
    Some(SimType::F125) => {
        // Check if DrivingDetector reported UdpActive on port 20777
        if conn.f1_udp_playable_received {
            // Emit AcStatus::Live with sim_type to server
            let msg = AgentMessage::GameStatusUpdate {
                pod_id: state.pod_id.clone(),
                ac_status: AcStatus::Live,
                sim_type: Some(SimType::F125),
            };
            if let Ok(json) = serde_json::to_string(&msg) {
                let _ = ws_tx.send(Message::Text(json.into())).await;
            }
            conn.launch_state = LaunchState::Live;
        }
    }
    Some(sim_type) => {
        // Process-based fallback for iRacing, LMU, EVO, WRC, etc.
        if let LaunchState::WaitingForLive { launched_at, .. } = &conn.launch_state {
            if launched_at.elapsed() >= Duration::from_secs(90) {
                let msg = AgentMessage::GameStatusUpdate {
                    pod_id: state.pod_id.clone(),
                    ac_status: AcStatus::Live,
                    sim_type: Some(sim_type),
                };
                if let Ok(json) = serde_json::to_string(&msg) {
                    let _ = ws_tx.send(Message::Text(json.into())).await;
                }
                conn.launch_state = LaunchState::Live;
            }
        }
    }
    None => {
        // No sim_type set — legacy behavior, AC path handles it
    }
}
```

**30s exit grace timer:**

7. Replace the IMMEDIATE `AcStatus::Off` emission on game exit with a grace timer start:
- When game process disappears (game_check detects no process) OR `AcStatus::Off` is detected from AC shared memory:
  - If crash recovery is NOT already active:
    ```rust
    conn.exit_grace_timer = Box::pin(tokio::time::sleep(Duration::from_secs(30)));
    conn.exit_grace_armed = true;
    conn.exit_grace_sim_type = conn.current_sim_type;
    ```
  - Do NOT emit `AcStatus::Off` yet.

8. Add a new arm in the select! loop for the grace timer:
```rust
_ = &mut conn.exit_grace_timer, if conn.exit_grace_armed => {
    conn.exit_grace_armed = false;
    // Grace period expired without relaunch — emit Off to server
    let msg = AgentMessage::GameStatusUpdate {
        pod_id: state.pod_id.clone(),
        ac_status: AcStatus::Off,
        sim_type: conn.exit_grace_sim_type,
    };
    if let Ok(json) = serde_json::to_string(&msg) {
        let _ = ws_tx.send(Message::Text(json.into())).await;
    }
    conn.exit_grace_sim_type = None;
    conn.current_sim_type = None;
    conn.loading_emitted = false;
    conn.f1_udp_playable_received = false;
}
```

9. Cancel the grace timer when crash recovery successfully relaunches:
- In the crash recovery success path (where `LaunchState::WaitingForLive` is set after relaunch):
```rust
conn.exit_grace_armed = false;
// Reset timer to far future
conn.exit_grace_timer = Box::pin(tokio::time::sleep(Duration::from_secs(86400)));
```
