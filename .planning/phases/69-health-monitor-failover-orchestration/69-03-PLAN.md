```javascript
function httpProbe(url, timeoutMs = 5000) {
  return new Promise((resolve) => {
    const req = http.get(url, { timeout: timeoutMs }, (res) => {
      resolve(res.statusCode >= 200 && res.statusCode < 500);
      res.resume();
    });
    req.on('error', () => resolve(false));
    req.on('timeout', () => { req.destroy(); resolve(false); });
  });
}
```

**Step 3: In the james_down event handler**, add the secondary watchdog timer:
```javascript
monitor.on('james_down', async (evt) => {
  // ... existing james_down handling (alertManager, logging, etc.) ...

  // Phase 69: Secondary watchdog -- if James still down after 5min total, check .23
  // HeartbeatMonitor fires james_down at 45s. We wait an additional 4min 15s = 255s
  // to reach the 5-minute total threshold.
  if (secondaryWatchdogTimer) clearTimeout(secondaryWatchdogTimer);
  secondaryWatchdogTimer = setTimeout(async () => {
    secondaryWatchdogTimer = null;

    // Re-check: is James still down?
    if (monitor.isUp) {
      console.log('[WATCHDOG] James recovered -- secondary watchdog cancelled');
      return;
    }

    // Probe server .23 via Tailscale
    const serverReachable = await httpProbe('http://100.71.226.83:8090/ping', 5000);
    if (serverReachable) {
      console.log('[WATCHDOG] Server .23 reachable via Tailscale -- not a server outage, skipping failover');
      return;
    }

    // Both James AND server .23 are down -- venue power outage scenario
    console.log('[WATCHDOG] James AND server .23 both down -- auto-activating cloud racecontrol');

    if (!failoverWatchdogCooldown.canSend()) {
      console.log('[WATCHDOG] Failover suppressed by cooldown');
      return;
    }

    // Bono IS the VPS -- start racecontrol locally via execFile (safe: no shell, hardcoded args)
    try {
      const { execFileSync } = await import('node:child_process');
      // Use pm2 restart to handle both "not running" and "already running" cases
      try {
        execFileSync('pm2', ['restart', 'racecontrol'], { cwd: '/root/racecontrol', timeout: 15000 });
      } catch (_restartErr) {
        // restart failed (process not found) -- try start
        execFileSync('pm2', ['start', 'racecontrol'], { cwd: '/root/racecontrol', timeout: 15000 });
      }
      console.log('[WATCHDOG] Cloud racecontrol started via pm2');

      // Wait for racecontrol to bind its port (poll health endpoint)
      let cloudReady = false;
      for (let i = 0; i < 6; i++) {
        await new Promise(r => setTimeout(r, 5000));
        cloudReady = await httpProbe('http://localhost:8080/api/v1/health', 3000);
        if (cloudReady) break;
      }

      if (!cloudReady) {
        console.error('[WATCHDOG] Cloud racecontrol did not become healthy after 30s');
        return;
      }

      // POST to local broadcast endpoint (Bono is localhost)
      const { httpPost } = await import('../shared/http-post.js');
      const result = await httpPost(
        'http://localhost:8080/api/v1/failover/broadcast',
        { target_url: 'ws://100.70.177.44:8080/ws/agent' },
        { headers: { 'x-terminal-secret': process.env.TERMINAL_SECRET || '' } }
      );
      console.log('[WATCHDOG] Broadcast result:', JSON.stringify(result));

      failoverWatchdogCooldown.recordSent();

      // Notify Uday via WhatsApp
      const text = `FAILOVER ACTIVATED (Bono watchdog) -- Server .23 AND James both unreachable. Pods switched to cloud (100.70.177.44). Time: ${new Date().toLocaleString('en-IN', { timeZone: 'Asia/Kolkata' })} IST. Likely venue power outage.`;
      if (alertManager?.handleNotification) {
        alertManager.handleNotification({ text });
      }
    } catch (err) {
      console.error('[WATCHDOG] Failed to activate cloud racecontrol:', err.message);
    }
  }, 255_000); // 4min 15s after james_down (which fires at 45s = 5min total)
});
```
