safe_remote_exec() {
  local host=$1 port=${2:-8090} cmd=$3 timeout=${4:-${DEFAULT_TIMEOUT:-10}}
  local tmpfile; tmpfile=$(mktemp)
  jq -n --arg cmd "$cmd" '{cmd: $cmd}' > "$tmpfile"
  local result
  result=$(curl -s -m "$timeout" -X POST "http://${host}:${port}/exec" \
    -H "Content-Type: application/json" -d @"$tmpfile" 2>/dev/null | tr -d '\r')
  rm -f "$tmpfile"
  printf '%s' "$result"
}

7. Increment DETECTOR_FINDINGS
8. `export -f detect_log_anomaly`
