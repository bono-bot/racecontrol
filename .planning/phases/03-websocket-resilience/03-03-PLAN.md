```typescript
  // Clear any pending disconnect timer -- we reconnected in time
  if (disconnectTimerRef.current !== null) {
    clearTimeout(disconnectTimerRef.current);
    disconnectTimerRef.current = null;
  }
  setConnected(true);
  console.log("[Kiosk] Connected to RaceControl");
};
```

**3. Modify socket.onclose (lines 272-276):**

Replace:
```typescript
socket.onclose = () => {
  setConnected(false);
  console.log("[Kiosk] Disconnected, retrying in 3s...");
  setTimeout(connect, 3000);
};
```
with:
```typescript
socket.onclose = () => {
  console.log("[Kiosk] Disconnected, retrying in 3s...");
  // Debounce UI update -- only show disconnected after 15s of confirmed absence
  // This prevents false "Disconnected" flashes during game launch CPU spikes
  if (disconnectTimerRef.current === null) {
    disconnectTimerRef.current = setTimeout(() => {
      setConnected(false);
      disconnectTimerRef.current = null;
      console.log("[Kiosk] 15s debounce expired -- marking disconnected");
    }, 15_000);
  }
  // Retry connection immediately (separate from UI debounce)
  setTimeout(connect, 3000);
};
```

Key points:
- `if (disconnectTimerRef.current === null)` prevents stacking multiple timers if onclose fires repeatedly during retries
- The 3s `setTimeout(connect, 3000)` is UNCHANGED -- reconnect attempts proceed immediately
- The UI debounce (15s) and reconnect retry (3s) are INDEPENDENT -- staff sees green while retries happen
- If a retry succeeds within 15s, onopen clears the timer and staff never sees a flash

**4. Update useEffect cleanup (lines 285-290):**

Replace:
```typescript
useEffect(() => {
  connect();
  return () => {
    ws.current?.close();
  };
}, [connect]);
```
with:
```typescript
useEffect(() => {
  connect();
  return () => {
    ws.current?.close();
    // Clean up debounce timer on unmount
    if (disconnectTimerRef.current !== null) {
      clearTimeout(disconnectTimerRef.current);
      disconnectTimerRef.current = null;
    }
  };
}, [connect]);
```

This prevents the timer from firing on an unmounted component (Pitfall 4 from research).

- NO sound alerts
