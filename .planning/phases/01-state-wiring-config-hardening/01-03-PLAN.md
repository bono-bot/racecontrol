```rust
#[derive(Debug, Deserialize)]
struct AgentConfig {
    pod: PodConfig,           // REQUIRED — [pod] section
    core: CoreConfig,         // REQUIRED — [core] section (url has default)
    #[serde(default)]
    wheelbase: WheelbaseConfig,
    #[serde(default)]
    telemetry_ports: TelemetryPortsConfig,
    #[serde(default)]
    games: GamesConfig,
    #[serde(default)]
    ai_debugger: AiDebuggerConfig,
    #[serde(default)]
    kiosk: KioskConfig,
}

#[derive(Debug, Deserialize)]
struct PodConfig {
    number: u32,              // REQUIRED — no default
    name: String,             // REQUIRED — no default
    sim: String,              // REQUIRED — no default
    #[serde(default = "default_sim_ip")]
    sim_ip: String,           // defaults to "127.0.0.1"
    #[serde(default = "default_sim_port")]
    sim_port: u16,            // defaults to 9996
}

#[derive(Debug, Deserialize)]
struct CoreConfig {
    #[serde(default = "default_core_url")]
    url: String,              // defaults to "ws://192.168.31.23:8080/ws/agent"
}
```
