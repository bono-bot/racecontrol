```rust
/// Rolling deploy: Pod 8 first (canary), then remaining pods.
/// Skips pods with active billing — sets them to WaitingSession.
/// If canary fails, halts and returns error.
pub async fn deploy_rolling(
    state: Arc<AppState>,
    binary_url: String,
) -> Result<(), String> {
    let canary_pod = "pod-8".to_string();
    let all_pods: Vec<String> = (1..=8).map(|n| format!("pod-{}", n)).collect();

    // Phase 1: Canary deploy to Pod 8
    tracing::info!("Rolling deploy: starting canary on Pod 8");
    deploy_pod(state.clone(), canary_pod.clone(), binary_url.clone()).await?;

    // Verify canary succeeded
    {
        let deploy_states = state.pod_deploy_states.read().await;
        match deploy_states.get(&canary_pod) {
            Some(DeployState::Complete) => {
                tracing::info!("Rolling deploy: canary Pod 8 healthy, proceeding to remaining pods");
            }
            Some(DeployState::Failed { reason }) => {
                return Err(format!("Canary Pod 8 failed: {}. Rolling deploy halted.", reason));
            }
            other => {
                return Err(format!("Canary Pod 8 in unexpected state: {:?}. Rolling deploy halted.", other));
            }
        }
    }

    // Phase 2: Deploy to remaining pods (1-7)
    for pod_id in all_pods.iter().filter(|p| *p != &canary_pod) {
        // Check for active billing session
        let has_active_session = {
            let timers = state.billing.active_timers.read().await;
            timers.contains_key(pod_id)
        };

        if has_active_session {
            tracing::info!("Rolling deploy: {} has active session, queuing", pod_id);
            {
                let mut deploy_states = state.pod_deploy_states.write().await;
                deploy_states.insert(pod_id.clone(), DeployState::WaitingSession);
            }
            let _ = state.dashboard_tx.send(DashboardEvent::DeployProgress {
                pod_id: pod_id.clone(),
                state: DeployState::WaitingSession,
                message: "Waiting for active session to end".to_string(),
                timestamp: Utc::now().to_rfc3339(),
            });
            // Store binary_url in pending_deploys for session-end hook
            {
                let mut pending = state.pending_deploys.write().await;
                pending.insert(pod_id.clone(), binary_url.clone());
            }
            continue;
        }

        // Deploy immediately
        if let Err(e) = deploy_pod(state.clone(), pod_id.clone(), binary_url.clone()).await {
            tracing::error!("Rolling deploy: {} failed: {}", pod_id, e);
            // Continue to next pod — don't halt for non-canary failures
        }
    }

    Ok(())
}
```

```rust
#[derive(Deserialize)]
struct RollingDeployRequest {
    binary_url: String,
}

async fn deploy_rolling_handler(
    State(state): State<Arc<AppState>>,
    Json(req): Json<RollingDeployRequest>,
) -> impl IntoResponse {
    // Check if any deploy is already in progress
    {
        let deploy_states = state.pod_deploy_states.read().await;
        let any_active = deploy_states.values().any(|s| !matches!(s, DeployState::Idle | DeployState::Complete | DeployState::Failed { .. }));
        if any_active {
            return (StatusCode::CONFLICT, Json(serde_json::json!({"error": "A deploy is already in progress"}))).into_response();
        }
    }

    let state_clone = state.clone();
    let url = req.binary_url.clone();
    tokio::spawn(async move {
        if let Err(e) = deploy::deploy_rolling(state_clone, url).await {
            tracing::error!("Rolling deploy failed: {}", e);
        }
    });

    (StatusCode::ACCEPTED, Json(serde_json::json!({"status": "rolling deploy started", "canary": "pod-8"}))).into_response()
}
```

```tsx
// DeployPanel: shows binary URL input, Deploy button, per-pod progress cards
// Uses Racing Point brand colors:
// - #E10600 (Racing Red) for failed
// - #1A1A1A (Asphalt Black) for background
// - Green (#22c55e) for complete
// - Yellow (#eab308) for in-progress states
// - Grey (#5A5A5A) for idle/waiting

export function DeployPanel({ deployStates, onDeploy }: DeployPanelProps) {
  const [binaryUrl, setBinaryUrl] = useState("");
  const [deploying, setDeploying] = useState(false);

  const handleDeploy = async () => {
    setDeploying(true);
    // Send DashboardCommand::DeployRolling via WebSocket
    onDeploy(binaryUrl);
  };

  return (
    <div className="space-y-4">
      <h2 className="text-lg font-semibold text-white" style={{ fontFamily: 'Enthocentric' }}>
        Deploy rc-agent
      </h2>
      <div className="flex gap-2">
        <input
          type="text"
          value={binaryUrl}
          onChange={(e) => setBinaryUrl(e.target.value)}
          placeholder="http://192.168.31.27:9998/rc-agent.exe"
          className="flex-1 bg-[#1A1A1A] border border-[#333333] rounded px-3 py-2 text-white"
        />
        <button
          onClick={handleDeploy}
          disabled={deploying || !binaryUrl}
          className="bg-[#E10600] text-white px-4 py-2 rounded disabled:opacity-50"
        >
          {deploying ? "Deploying..." : "Deploy All"}
        </button>
      </div>
      <div className="grid grid-cols-4 gap-2">
        {[1,2,3,4,5,6,7,8].map(n => {
          const podId = `pod-${n}`;
          const state = deployStates.get(podId) || "idle";
          return (
            <DeployPodCard key={podId} podId={podId} state={state} isCanary={n === 8} />
          );
        })}
      </div>
    </div>
  );
}
```
