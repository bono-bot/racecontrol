3. **HTML5 Drag and Drop handlers** (add to each .cam tile during buildGrid):

   `dragstart` handler:
   ```
   tile.addEventListener('dragstart', function(e) {
     dragSrcChannel = parseInt(this.dataset.channel);
     this.classList.add('dragging');
     e.dataTransfer.effectAllowed = 'move';
     e.dataTransfer.setData('text/plain', String(dragSrcChannel));
   });
   ```

   `dragover` handler (MUST call e.preventDefault() or drop never fires):
   ```
   tile.addEventListener('dragover', function(e) {
     e.preventDefault();
     e.dataTransfer.dropEffect = 'move';
     this.classList.add('drag-over');
   });
   ```

   `dragleave` handler:
   ```
   tile.addEventListener('dragleave', function(e) {
     this.classList.remove('drag-over');
   });
   ```

   `dragend` handler:
   ```
   tile.addEventListener('dragend', function(e) {
     this.classList.remove('dragging');
     // Clean up any lingering drag-over highlights
     document.querySelectorAll('.drag-over').forEach(function(el) {
       el.classList.remove('drag-over');
     });
   });
   ```

   `drop` handler:
   ```
   tile.addEventListener('drop', function(e) {
     e.preventDefault();
     this.classList.remove('drag-over');
     var targetChannel = parseInt(this.dataset.channel);
     if (dragSrcChannel === null || dragSrcChannel === targetChannel) return;

     // Find source and target indices in cameras array
     var srcIdx = cameras.findIndex(function(c) { return c.nvr_channel === dragSrcChannel; });
     var tgtIdx = cameras.findIndex(function(c) { return c.nvr_channel === targetChannel; });
     if (srcIdx === -1 || tgtIdx === -1) return;

     // Reorder: remove source, insert at target position
     var moved = cameras.splice(srcIdx, 1)[0];
     cameras.splice(tgtIdx, 0, moved);

     // Rebuild grid to reflect new order (preserves zone grouping)
     buildGrid();
     // Auto-save on drop (per user decision: no save button needed)
     saveLayout();
     dragSrcChannel = null;
   });
   ```

4. **Modify saveLayout()** to include the current camera order:

   ```
   function saveLayout() {
     fetch('/api/v1/cameras/layout', {
       method: 'PUT',
       headers: { 'Content-Type': 'application/json' },
       body: JSON.stringify({
         grid_mode: currentMode,
         camera_order: cameras.map(function(c) { return c.nvr_channel; }).filter(Boolean),
         zone_filter: null
       })
     }).catch(function(err) { console.warn('Layout save failed:', err); });
   }
   ```

5. **Modify fetchLayout()** to apply saved camera_order:
   After fetching layout, if `camera_order` is a non-empty array, reorder the `cameras` array to match:
   ```
   if (data.camera_order && data.camera_order.length > 0) {
     var orderMap = {};
     data.camera_order.forEach(function(ch, i) { orderMap[ch] = i; });
     cameras.sort(function(a, b) {
       var aIdx = (a.nvr_channel in orderMap) ? orderMap[a.nvr_channel] : 9999;
       var bIdx = (b.nvr_channel in orderMap) ? orderMap[b.nvr_channel] : 9999;
       return aIdx - bIdx;
     });
   }
   ```
   Then call `buildGrid()` to re-render with the saved order.
