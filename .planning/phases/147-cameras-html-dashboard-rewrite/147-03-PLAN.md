<!-- go2rtc WebRTC signaling protocol (from research): -->

WebSocket URL: ws://192.168.31.27:1984/api/ws?src={stream_name}
Stream names: ch1, ch2, ..., ch13 (matching go2rtc.yaml)

Protocol messages (JSON over WebSocket):
1. Browser sends: { "type": "webrtc/offer", "value": sdpOffer }
2. go2rtc responds: { "type": "webrtc/answer", "value": sdpAnswer }
3. Either side sends: { "type": "webrtc/candidate", "value": candidateJSON }

Flow:
1. Open WebSocket to ws://192.168.31.27:1984/api/ws?src=ch{N}
2. Create RTCPeerConnection
3. Add transceiver for video (recvonly)
4. Create offer, set local description
5. Send offer via WebSocket
6. Receive answer, set remote description
7. Exchange ICE candidates via WebSocket
8. Video track arrives on pc.ontrack -> attach to video element

Teardown:
- pc.close() stops all tracks and connections
- ws.close() disconnects signaling
- Must call on: fullscreen close, camera switch, page beforeunload, visibilitychange

Camera object shape (from plan 01):
{ name, display_name, display_order, role, zone, nvr_channel, stream_url, status }

Stream name for WebRTC: "ch" + camera.nvr_channel (e.g., "ch1", "ch13")

4. **connectWebRTC(streamName, onTrack, onStatus) -- core signaling function:**
   ```
   function connectWebRTC(streamName, onTrack, onStatus) {
     var ws = new WebSocket(GO2RTC_WS + '?src=' + streamName);
     var pc = new RTCPeerConnection({
       iceServers: [{ urls: 'stun:stun.l.google.com:19302' }]
     });

     pc.addTransceiver('video', { direction: 'recvonly' });
     pc.addTransceiver('audio', { direction: 'recvonly' });

     pc.ontrack = function(e) {
       if (onTrack) onTrack(e.streams[0]);
     };

     pc.onicecandidate = function(e) {
       if (e.candidate && ws.readyState === WebSocket.OPEN) {
         ws.send(JSON.stringify({
           type: 'webrtc/candidate',
           value: e.candidate.toJSON()
         }));
       }
     };

     pc.onconnectionstatechange = function() {
       if (onStatus) onStatus(pc.connectionState);
     };

     ws.onopen = function() {
       pc.createOffer().then(function(offer) {
         return pc.setLocalDescription(offer);
       }).then(function() {
         ws.send(JSON.stringify({
           type: 'webrtc/offer',
           value: pc.localDescription.sdp
         }));
       }).catch(function(err) {
         console.error('WebRTC offer failed:', err);
         if (onStatus) onStatus('failed');
       });
     };

     ws.onmessage = function(ev) {
       var msg = JSON.parse(ev.data);
       if (msg.type === 'webrtc/answer') {
         pc.setRemoteDescription(new RTCSessionDescription({
           type: 'answer',
           sdp: msg.value
         })).catch(function(err) {
           console.error('setRemoteDescription failed:', err);
         });
       } else if (msg.type === 'webrtc/candidate') {
         pc.addIceCandidate(new RTCIceCandidate(msg.value)).catch(function(err) {
           console.debug('addIceCandidate failed:', err);
         });
       }
     };

     ws.onerror = function() {
       if (onStatus) onStatus('failed');
     };

     return { pc: pc, ws: ws };
   }
   ```

5. **openFullscreen(camera) -- replaces the placeholder:**
   ```
   function openFullscreen(camera) {
     var streamName = 'ch' + camera.nvr_channel;

     // Tear down any existing connection (singleton enforcement)
     teardownRtc();

     // Show fullscreen overlay
     var fsEl = document.getElementById('fs');
     fsEl.classList.add('visible');
     document.getElementById('fs-name').textContent = camera.display_name;

     var dotEl = document.getElementById('fs-dot');
     dotEl.className = 'fs-status-dot connecting';

     var loadingEl = document.getElementById('fs-loading');
     loadingEl.style.display = 'block';

     var videoEl = document.getElementById('fs-video');
     videoEl.srcObject = null;

     // Check if we have a pre-warmed connection for this camera
     if (preWarmChannel === camera.nvr_channel && preWarmPc && preWarmWs) {
       activePc = preWarmPc;
       activeWs = preWarmWs;
       preWarmPc = null;
       preWarmWs = null;
       preWarmChannel = null;

       // Wire up the existing pre-warmed connection
       activePc.ontrack = function(e) {
         videoEl.srcObject = e.streams[0];
         loadingEl.style.display = 'none';
         dotEl.className = 'fs-status-dot connected';
       };
       activePc.onconnectionstatechange = function() {
         updateFsStatus(activePc.connectionState, dotEl, loadingEl);
       };
       // If tracks already arrived during pre-warm
       var receivers = activePc.getReceivers();
       for (var i = 0; i < receivers.length; i++) {
         if (receivers[i].track && receivers[i].track.kind === 'video') {
           var stream = new MediaStream([receivers[i].track]);
           videoEl.srcObject = stream;
           loadingEl.style.display = 'none';
           dotEl.className = 'fs-status-dot connected';
           break;
         }
       }
       activeCamera = camera;
     } else {
       // Clean up any unrelated pre-warm
       teardownPreWarm();

       // Create fresh connection
       var conn = connectWebRTC(streamName, function(stream) {
         videoEl.srcObject = stream;
         loadingEl.style.display = 'none';
         dotEl.className = 'fs-status-dot connected';
       }, function(state) {
         updateFsStatus(state, dotEl, loadingEl);
       });
       activePc = conn.pc;
       activeWs = conn.ws;
       activeCamera = camera;
     }

     // Auto-hide controls after 3 seconds
     startControlsAutoHide();

     // Show fallback snapshot as background while loading
     if (camera.nvr_channel) {
       videoEl.poster = '/api/v1/cameras/nvr/' + camera.nvr_channel + '/snapshot?t=' + Date.now();
     }
   }
   ```

6. **updateFsStatus(state, dotEl, loadingEl):**
   ```
   function updateFsStatus(state, dotEl, loadingEl) {
     if (state === 'connected') {
       dotEl.className = 'fs-status-dot connected';
       loadingEl.style.display = 'none';
     } else if (state === 'failed' || state === 'disconnected' || state === 'closed') {
       dotEl.className = 'fs-status-dot failed';
       loadingEl.style.display = 'none';
       // Show fallback message
       showFallbackMessage();
     }
   }
   ```
