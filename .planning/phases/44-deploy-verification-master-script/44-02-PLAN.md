**Helper: run_phase()**
```bash
run_phase() {
    local name="$1"; shift
    local logfile="$RESULTS_DIR/${name}.log"
    echo ""
    echo "========================================"
    echo "  Phase: ${name}"
    echo "========================================"
    if "$@" 2>&1 | tee "$logfile"; then
        return 0
    else
        return $?
    fi
}
```
NOTE: Use `"$@"` not `${cmd[@]}` to avoid bash 3 issues. The pipe to `tee` captures log while still showing output. Use `set -o pipefail` at top so tee does not mask exit codes.

Actually, `tee` in a pipe will mask the exit code on bash without pipefail. Since we want per-phase exit codes, use a different approach:
```bash
run_phase() {
    local name="$1"; shift
    local logfile="$RESULTS_DIR/${name}.log"
    echo ""
    echo "========================================"
    echo "  Phase: ${name}"
    echo "========================================"
    "$@" > >(tee "$logfile") 2>&1
    return $?
}
```
Or simpler: run the command, capture exit code, then just redirect to log:
```bash
run_phase() {
    local name="$1"; shift
    local logfile="$RESULTS_DIR/${name}.log"
    echo ""
    echo "========================================"
    echo "  Phase: ${name}"
    echo "========================================"
    "$@" 2>&1 | tee "$logfile"
    return "${PIPESTATUS[0]}"
}
```
Use `PIPESTATUS[0]` to get the exit code of the command, not tee. This requires `set -o pipefail` NOT be set at the run-all.sh level (common.sh does not set it, and run-all.sh should not either -- we want to capture per-phase exit codes manually).

**Phase 1: Preflight**
```bash
echo ""
echo "============================================================"
echo "  RaceControl E2E Test Suite"
echo "  $(date '+%Y-%m-%d %H:%M:%S IST')"
echo "============================================================"

# Phase 1a: Smoke
run_phase "smoke" bash "$SCRIPT_DIR/smoke.sh"
SMOKE_EXIT="${PIPESTATUS[0]}"

if [ "$SMOKE_EXIT" -ne 0 ]; then
    echo ""
    echo "PREFLIGHT FAILED -- smoke tests returned ${SMOKE_EXIT} failures"
    echo "Aborting remaining phases."
    PREFLIGHT_EXIT=$SMOKE_EXIT
    PREFLIGHT_STATUS="FAIL"
    TOTAL_FAIL=$((TOTAL_FAIL + SMOKE_EXIT))
    # Skip to summary
else
    PREFLIGHT_STATUS="PASS"

    # Phase 1b: Cross-process
    run_phase "cross-process" bash "$SCRIPT_DIR/cross-process.sh"
    CROSS_EXIT="${PIPESTATUS[0]}"
    if [ "$CROSS_EXIT" -ne 0 ]; then
        PREFLIGHT_EXIT=$CROSS_EXIT
        PREFLIGHT_STATUS="FAIL"
        TOTAL_FAIL=$((TOTAL_FAIL + CROSS_EXIT))
        echo ""
        echo "PREFLIGHT FAILED -- cross-process returned ${CROSS_EXIT} failures"
        echo "Aborting remaining phases."
    else
        PREFLIGHT_STATUS="PASS"
    fi
fi
```

If preflight fails, skip to summary (do NOT exit -- still write summary.json).
