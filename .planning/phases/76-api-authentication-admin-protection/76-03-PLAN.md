    **Step 2: Create require_service_key middleware**

    In `crates/rc-agent/src/remote_ops.rs`, add a middleware function:

    ```rust
    use subtle::ConstantTimeEq;

    async fn require_service_key(
        req: axum::extract::Request,
        next: axum::middleware::Next,
    ) -> Result<axum::response::Response, StatusCode> {
        let expected = std::env::var("RCAGENT_SERVICE_KEY").unwrap_or_default();

        // Permissive mode: no key configured = allow all (Phase A of expand-migrate-contract)
        if expected.is_empty() {
            return Ok(next.run(req).await);
        }

        let provided = req.headers()
            .get("x-service-key")
            .and_then(|v| v.to_str().ok())
            .unwrap_or("");

        // Constant-time comparison to prevent timing attacks
        if expected.as_bytes().ct_eq(provided.as_bytes()).into() {
            Ok(next.run(req).await)
        } else {
            Err(StatusCode::UNAUTHORIZED)
        }
    }
    ```
