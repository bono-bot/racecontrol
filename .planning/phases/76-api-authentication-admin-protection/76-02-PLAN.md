    **Step 3: Create admin.rs module**

    Create `crates/racecontrol/src/auth/admin.rs`:

    ```rust
    use argon2::{Argon2, PasswordHash, PasswordHasher, PasswordVerifier};
    use argon2::password_hash::SaltString;
    use rand::rngs::OsRng;
    use axum::{extract::State, Json, http::StatusCode};
    use serde::{Deserialize, Serialize};
    use std::sync::Arc;
    use crate::state::AppState;

    #[derive(Deserialize)]
    pub struct AdminLoginRequest {
        pub pin: String,
    }

    #[derive(Serialize)]
    pub struct AdminLoginResponse {
        pub token: String,
        pub expires_in: u64,
    }

    /// Hash a PIN using Argon2id. Used for initial setup (CLI or first-run).
    pub fn hash_admin_pin(pin: &str) -> Result<String, String> {
        let salt = SaltString::generate(&mut OsRng);
        let argon2 = Argon2::default();
        let hash = argon2.hash_password(pin.as_bytes(), &salt)
            .map_err(|e| format!("Argon2 hash error: {}", e))?;
        Ok(hash.to_string())
    }

    /// Verify a PIN against an argon2 hash string.
    pub fn verify_admin_pin(pin: &str, hash_str: &str) -> bool {
        let parsed = match PasswordHash::new(hash_str) {
            Ok(h) => h,
            Err(_) => return false,
        };
        Argon2::default().verify_password(pin.as_bytes(), &parsed).is_ok()
    }

    /// POST /api/v1/auth/admin-login
    /// Validates admin PIN against argon2 hash, returns staff JWT with 12h expiry.
    pub async fn admin_login(
        State(state): State<Arc<AppState>>,
        Json(body): Json<AdminLoginRequest>,
    ) -> Result<Json<AdminLoginResponse>, StatusCode> {
        let pin_hash = state.config.auth.admin_pin_hash.as_ref()
            .ok_or(StatusCode::SERVICE_UNAVAILABLE)?;

        let pin = body.pin.clone();
        let hash = pin_hash.clone();

        // Argon2 is CPU-intensive (100-500ms) -- run on blocking thread
        let valid = tokio::task::spawn_blocking(move || {
            verify_admin_pin(&pin, &hash)
        })
        .await
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

        if !valid {
            return Err(StatusCode::UNAUTHORIZED);
        }

        // Create staff JWT with 12-hour expiry (ADMIN-03)
        let jwt_secret = &state.config.auth.jwt_secret;
        let claims = super::middleware::StaffClaims {
            sub: "admin".to_string(),
            role: "staff".to_string(),
            exp: (chrono::Utc::now() + chrono::Duration::hours(12)).timestamp() as usize,
            iat: chrono::Utc::now().timestamp() as usize,
        };

        let token = jsonwebtoken::encode(
            &jsonwebtoken::Header::default(),
            &claims,
            &jsonwebtoken::EncodingKey::from_secret(jwt_secret.as_bytes()),
        ).map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

        Ok(Json(AdminLoginResponse {
            token,
            expires_in: 43200, // 12 hours in seconds
        }))
    }
    ```

    NOTE: If Plan 01 middleware.rs does not exist yet when this executes, create a minimal `StaffClaims` struct locally or in middleware.rs yourself. The executor for Plan 01 will handle any merge.
