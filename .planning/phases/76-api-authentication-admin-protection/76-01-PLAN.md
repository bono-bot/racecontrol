    ```rust
    pub async fn require_staff_jwt_permissive(
        State(state): State<Arc<AppState>>,
        mut req: Request,
        next: Next,
    ) -> Response {
        match extract_staff_claims(&state, &req) {
            Ok(claims) => { req.extensions_mut().insert(claims); }
            Err(_) => {
                tracing::warn!(
                    path = %req.uri().path(),
                    method = %req.method(),
                    "Unauthenticated staff request (permissive mode)"
                );
            }
        }
        next.run(req).await
    }
    ```
