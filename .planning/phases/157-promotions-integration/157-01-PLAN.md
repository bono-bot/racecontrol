pub async fn list_active_promos(
    State(state): State<Arc<AppState>>,
) -> Result<(StatusCode, Json<Vec<ActivePromo>>), (StatusCode, Json<serde_json::Value>)> {
    // Fetch all is_active promos
    let promos = sqlx::query_as::<_, CafePromo>(
        "SELECT id, name, promo_type, config, is_active, start_time, end_time, stacking_group, created_at, updated_at
         FROM cafe_promos WHERE is_active = 1 ORDER BY created_at DESC",
    )
    .fetch_all(&state.db)
    .await
    .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, Json(json!({"error": e.to_string()}))))?;

    // Filter to time-window matches in IST and build ActivePromo list
    let now_ist = ist_now_hhmm();
    let active: Vec<ActivePromo> = promos
        .into_iter()
        .filter_map(|p| {
            // Check time window: if both start and end are set, must be within window
            if let (Some(ref start), Some(ref end)) = (&p.start_time, &p.end_time) {
                if !time_in_window(&now_ist, start, end) {
                    return None;
                }
            }
            let config: serde_json::Value =
                serde_json::from_str(&p.config).unwrap_or(serde_json::Value::Object(Default::default()));
            let time_label = match (&p.start_time, &p.end_time) {
                (Some(_), Some(ref end)) => Some(format!("Active until {}", fmt_hhmm(end))),
                _ => None,
            };
            Some(ActivePromo {
                id: p.id,
                name: p.name,
                promo_type: p.promo_type,
                config,
                stacking_group: p.stacking_group,
                time_label,
            })
        })
        .collect();

    Ok((StatusCode::OK, Json(active)))
}

/// Evaluate which promo (if any) applies to this cart and return the best discount.
/// cart_items: Vec<(item_id, quantity)>
/// active_promos: slice from list_active_promos_inner (same filtering)
pub fn evaluate_promos(
    cart_items: &[(String, i64)],  // (item_id, quantity)
    active_promos: &[ActivePromo],
) -> PromoEvalResult {
    // Build a map of item_id -> quantity in cart for fast lookup
    let cart_map: std::collections::HashMap<&str, i64> =
        cart_items.iter().map(|(id, qty)| (id.as_str(), *qty)).collect();

    // Collect one candidate per stacking_group (or per promo if no group)
    // Key = stacking_group or promo id; Value = best (discount_paise, promo)
    let mut group_best: std::collections::HashMap<String, (i64, &ActivePromo)> =
        std::collections::HashMap::new();

    for promo in active_promos {
        let discount = calc_promo_discount(promo, &cart_map);
        if discount <= 0 {
            continue;
        }
        let key = promo
            .stacking_group
            .clone()
            .unwrap_or_else(|| promo.id.clone());
        let entry = group_best.entry(key).or_insert((0, promo));
        if discount > entry.0 {
            *entry = (discount, promo);
        }
    }

    // Sum all group winners (groups cannot combine with each other per PROMO-04:
    // stacking_group = exclusive within group, different groups can stack)
    // Per CONTEXT.md: "promos in same stacking_group are exclusive (pick best discount)"
    // Different groups CAN combine — sum them.
    let mut total_discount: i64 = 0;
    let mut applied_id: Option<String> = None;
    let mut applied_name: Option<String> = None;

    // If multiple groups apply, pick only the single largest discount for simplicity
    // (Claude's discretion — CONTEXT.md does not mandate multi-group stacking in v1)
    if let Some((discount, promo)) = group_best.values().max_by_key(|(d, _)| *d) {
        total_discount = *discount;
        applied_id = Some(promo.id.clone());
        applied_name = Some(promo.name.clone());
    }

    PromoEvalResult {
        applied_promo_id: applied_id,
        promo_name: applied_name,
        discount_paise: total_discount,
    }
}

/// Calculate the discount in paise that a single promo gives for this cart.
/// Returns 0 if the promo conditions are not met.
fn calc_promo_discount(
    promo: &ActivePromo,
    cart_map: &std::collections::HashMap<&str, i64>,
) -> i64 {
    match promo.promo_type.as_str() {
        "combo" => {
            // config: { "items": [{"item_id": "...", "quantity": 1}], "bundle_price_paise": 14900 }
            // OR config: { "items": [...], "discount_percent": 10 }
            // Conditions: all required items present with sufficient qty
            let required = match promo.config.get("items").and_then(|v| v.as_array()) {
                Some(arr) => arr.clone(),
                None => return 0,
            };
            let mut gross: i64 = 0;
            for req_item in &required {
                let item_id = match req_item.get("item_id").and_then(|v| v.as_str()) {
                    Some(id) => id,
                    None => return 0,
                };
                let req_qty = req_item.get("quantity").and_then(|v| v.as_i64()).unwrap_or(1);
                let cart_qty = *cart_map.get(item_id).unwrap_or(&0);
                if cart_qty < req_qty {
                    return 0;  // condition not met
                }
                let unit_price = req_item.get("unit_price_paise").and_then(|v| v.as_i64()).unwrap_or(0);
                gross += unit_price * req_qty;
            }
            if let Some(bundle_price) = promo.config.get("bundle_price_paise").and_then(|v| v.as_i64()) {
                return (gross - bundle_price).max(0);
            }
            if let Some(pct) = promo.config.get("discount_percent").and_then(|v| v.as_i64()) {
                return (gross * pct / 100).max(0);
            }
            0
        }
        "happy_hour" => {
            // config: { "discount_percent": 20, "target_item_ids": ["id1", "id2"] }
            // OR target_item_ids absent = applies to entire cart
            // Time window already enforced by list_active_promos
            let pct = match promo.config.get("discount_percent").and_then(|v| v.as_i64()) {
                Some(p) if p > 0 => p,
                _ => return 0,
            };
            let targets = promo.config.get("target_item_ids").and_then(|v| v.as_array());
            let gross: i64 = match targets {
                Some(ids) => {
                    let id_set: std::collections::HashSet<&str> =
                        ids.iter().filter_map(|v| v.as_str()).collect();
                    cart_map
                        .iter()
                        .filter(|(id, _)| id_set.contains(*id))
                        .map(|(_, &qty)| {
                            // We don't have unit prices here — happy hour discount
                            // is computed post-total in place_cafe_order_inner using
                            // verified_items prices. Return qty as placeholder;
                            // actual price multiplication done in caller.
                            // IMPORTANT: evaluate_promos receives cart_items without prices.
                            // For happy_hour with target_item_ids, return 0 here;
                            // caller must pass pre-computed gross for those items.
                            // Design decision: happy_hour discount computed on total_paise
                            // when target_item_ids is absent; item-targeted happy_hour
                            // requires unit prices — caller passes gross via config workaround.
                            qty  // placeholder, see note in action
                        })
                        .sum()
                }
                None => 0, // signal: apply to total
            };
            // If no target items specified, discount applies to entire order total
            // The caller (place_cafe_order_inner) must handle this case:
            // after evaluate_promos returns, if promo_type == "happy_hour" and
            // discount_paise == 0, check config for discount_percent and apply to total_paise.
            // See Task 3 for full integration.
            let _ = gross;
            0  // placeholder — actual computation in Task 3 integration
        }
        "gaming_bundle" => {
            // Not fully evaluatable without DB (need active billing session).
            // Return 0 — gaming_bundle evaluation happens in place_cafe_order_inner
            // where the DB is available. Listed in active promos for display only.
            0
        }
        _ => 0,
    }
}

pub fn evaluate_promos(
    cart_items: &[(String, i64)],
    active_promos: &[ActivePromo],
    total_paise: i64,  // add this param — needed for happy_hour %
) -> PromoEvalResult {
    let cart_map: std::collections::HashMap<&str, i64> =
        cart_items.iter().map(|(id, qty)| (id.as_str(), *qty)).collect();

    let mut group_best: std::collections::HashMap<String, (i64, String, String)> =
        std::collections::HashMap::new();  // group_key -> (discount_paise, promo_id, promo_name)

    for promo in active_promos {
        let discount = calc_promo_discount(promo, &cart_map, total_paise);
        if discount <= 0 {
            continue;
        }
        let key = promo.stacking_group.clone().unwrap_or_else(|| promo.id.clone());

        if discount > entry.0 {
            *entry = (discount, promo.id.clone(), promo.name.clone());
        }
    }

    if let Some((discount, id, name)) = group_best.values().max_by_key(|(d, _, _)| *d) {
        PromoEvalResult {
            applied_promo_id: Some(id.clone()),
            promo_name: Some(name.clone()),
            discount_paise: *discount,

    } else {
        PromoEvalResult::default()
    }
}

```rust
/// Returns current IST time as "HH:MM" string.
fn ist_now_hhmm() -> String {
    // IST = UTC+5:30 = UTC + 19800 seconds
    let now_utc = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs();
    let ist_secs = now_utc + 19800;
    let hours = (ist_secs / 3600) % 24;
    let minutes = (ist_secs % 3600) / 60;
    format!("{:02}:{:02}", hours, minutes)
}

/// Returns true if `now` (HH:MM) is within [start, end) window.
/// Handles overnight wrap (e.g. 22:00 to 02:00) correctly.
fn time_in_window(now: &str, start: &str, end: &str) -> bool {
    if now >= start && start < end {
        now < end
    } else if start >= end {
        // overnight: e.g. 22:00 to 02:00
        now >= start || now < end
    } else {
        false
    }
}

/// Format "15:00" as "3:00 PM".
fn fmt_hhmm(hhmm: &str) -> String {
    let parts: Vec<&str> = hhmm.splitn(2, ':').collect();
    if parts.len() != 2 { return hhmm.to_string(); }
    let h: u32 = parts[0].parse().unwrap_or(0);
    let m = parts[1];
    let period = if h < 12 { "AM" } else { "PM" };
    let h12 = match h % 12 { 0 => 12, v => v };
    format!("{}:{} {}", h12, m, period)
}
```
