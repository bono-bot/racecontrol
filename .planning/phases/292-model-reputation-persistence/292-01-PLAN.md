<objective>
Replace the ephemeral in-memory model reputation store with a SQLite-backed persistence layer. On boot, past accuracy counts and demotion/promotion decisions are restored from `mesh_kb.db`. The daily reputation sweep queries the authoritative 7-day window from `model_evaluations` (Phase 290 data) rather than the in-memory counters that reset on restart.

Purpose: MREP-01/02/03. Without persistence, every restart wipes reputation history — a poorly-performing model that was demoted comes back fresh. With persistence, the roster self-curates durably.
Output: `model_reputation_store.rs` (new file), `model_reputation.rs` (updated sweep logic), `main.rs` (boot wiring).
</objective>

for (model_id, (correct, total)) in &per_model {
    let accuracy = if *total > 0 { *correct as f64 / *total as f64 } else { 0.5 };
    let total_runs = *total;

    if accuracy < DEMOTE_ACCURACY_THRESHOLD && total_runs >= DEMOTE_MIN_RUNS {
        tracing::warn!(
            target: LOG_TARGET,
            model = %model_id,
            accuracy = accuracy,
            runs = total_runs,
            "REP-01: Demoting model — accuracy below {}% over 7-day window",
            (DEMOTE_ACCURACY_THRESHOLD * 100.0) as u32
        );
        mma_engine::demote_model(model_id);
        // MREP-02: persist demotion
        if let Ok(store) = rep_store.lock() {
            if let Err(e) = store.save_demotion(model_id) {
                tracing::warn!(target: LOG_TARGET, model = %model_id, error = %e, "Failed to persist demotion");
            }

        let _ = fleet_tx.send(FleetEvent::ModelReputationChange {
            model_id: model_id.clone(),
            action: "demoted".to_string(),
            accuracy,
            total_runs,
            timestamp: Utc::now(),
        });

    // MREP-03: Promote high-accuracy models
    if accuracy > PROMOTE_ACCURACY_THRESHOLD && total_runs >= PROMOTE_MIN_RUNS {

            accuracy = accuracy,
