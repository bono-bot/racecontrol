    ```rust
    /// Update success/failure outcome for a solution.
    /// Recalculates confidence = success_count / (success_count + fail_count).
    /// Also updates updated_at to current IST timestamp and increments version.
    pub fn record_outcome(&self, id: &str, succeeded: bool) -> anyhow::Result<()> {
        // Fetch current counts
        let result: Result<(i64, i64), _> = self.conn.query_row(
            "SELECT success_count, fail_count FROM solutions WHERE id = ?1",
            params![id],
            |row| Ok((row.get(0)?, row.get(1)?)),
        );

        let (success_count, fail_count) = match result {
            Ok(counts) => counts,
            Err(rusqlite::Error::QueryReturnedNoRows) => {
                tracing::debug!(target: LOG_TARGET, id = id, "record_outcome: solution id not found, skipping");
                return Ok(());
            }
            Err(e) => return Err(anyhow::anyhow!("record_outcome fetch error: {}", e)),
        };

        let (new_success, new_fail) = if succeeded {
            (success_count + 1, fail_count)
        } else {
            (success_count, fail_count + 1)
        };

        let total = new_success + new_fail;
        let new_confidence = if total > 0 {
            new_success as f64 / total as f64
        } else {
            // Shouldn't happen — keep current value by re-reading; use 0.0 as safe fallback
            0.0
        };

        let updated_at = ist_now();

        self.conn.execute(
            "UPDATE solutions SET
                success_count = ?1,
                fail_count = ?2,
                confidence = ?3,
                updated_at = ?4,
                version = version + 1
             WHERE id = ?5",
            params![new_success, new_fail, new_confidence, updated_at, id],
        )?;

        tracing::debug!(
            target: LOG_TARGET,
            id = id,
            succeeded = succeeded,
            new_confidence = new_confidence,
            "Solution outcome recorded"
        );
        Ok(())
    }

    /// Archive (delete) solutions that have not been updated within their TTL window.
    /// Uses julianday arithmetic — SQLite built-in, no external date parsing needed.
    /// Returns the count of archived solutions.
    pub fn archive_expired_solutions(&self) -> anyhow::Result<usize> {
        let count = self.conn.execute(
            "DELETE FROM solutions WHERE julianday('now') - julianday(updated_at) > ttl_days",
            [],
        )?;
        if count > 0 {
            tracing::info!(
                target: LOG_TARGET,
                archived = count,
                "TTL expiry: archived expired solutions from local KB"
            );
        }
        Ok(count)
    }

    /// Compute a stable SHA256 hash for a DiagnosticTrigger.
    /// The hash is derived from normalize_problem_key — same trigger class → same hash.
    /// Used as the problem_hash column in the solutions table.
    pub fn problem_hash_from_trigger(trigger: &DiagnosticTrigger) -> String {
        use sha2::{Sha256, Digest};
        let key = normalize_problem_key(trigger);
        let hash = Sha256::digest(key.as_bytes());
        hex::encode(hash)
    }

    fn ist_now() -> String {
        use chrono::{FixedOffset, Utc};
        let ist = FixedOffset::east_opt(5 * 3600 + 30 * 60)
            .unwrap_or_else(|| FixedOffset::east_opt(0).expect("UTC offset 0 valid"));
        Utc::now().with_timezone(&ist).format("%Y-%m-%dT%H:%M:%S%:z").to_string()

    fn tier2_kb_lookup(event: &DiagnosticEvent) -> TierResult {
        let problem_key = crate::knowledge_base::normalize_problem_key(&event.trigger);
        let problem_hash = crate::knowledge_base::problem_hash_from_trigger(&event.trigger);

        tracing::debug!(
            target: LOG_TARGET,
            tier = 2u8,
            problem_key = %problem_key,
            problem_hash = %problem_hash,
            "Tier 2: querying local KB"
        );

        // Open the local KB — if it doesn't exist yet (first run), treat as no match
        let kb = match crate::knowledge_base::KnowledgeBase::open(crate::knowledge_base::KB_PATH) {
            Ok(kb) => kb,
            Err(e) => {
                tracing::debug!(
                    target: LOG_TARGET,
                    error = %e,
                    "Tier 2: KB unavailable — treating as no match"
                );
                return TierResult::NotApplicable { tier: 2 };
            }
        };

        match kb.lookup(&problem_hash) {
            Ok(Some(solution)) => {
                tracing::info!(
                    target: LOG_TARGET,
                    tier = 2u8,
                    problem_key = %problem_key,
                    root_cause = %solution.root_cause,
                    confidence = solution.confidence,
                    fix_type = %solution.fix_type,
                    "Tier 2: KB hit — applying known solution"
                );
                TierResult::Fixed {
                    tier: 2,
                    action: format!(
                        "KB match (confidence={:.2}): {} — fix: {}",
                        solution.confidence,
                        solution.root_cause,
                        solution.fix_action
                    ),
                }
            }
            Ok(None) => {
                tracing::debug!(
                    target: LOG_TARGET,
                    tier = 2u8,
                    problem_key = %problem_key,
                    "Tier 2: KB miss — no high-confidence solution found"
                );
                TierResult::NotApplicable { tier: 2 }
            }
            Err(e) => {
                tracing::warn!(
                    target: LOG_TARGET,
                    tier = 2u8,
                    error = %e,
                    problem_key = %problem_key,
                    "Tier 2: KB lookup error — treating as no match"
                );
                TierResult::NotApplicable { tier: 2 }
            }
        }
    }
    ```
