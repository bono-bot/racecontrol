const F1_2010_POINTS: [i64; 10] = [25, 18, 15, 12, 10, 8, 6, 4, 2, 1];

fn f1_points_for_position(position: i64, dnf: bool) -> i64 {
    if dnf || position < 1 || position > 10 { return 0; }
    F1_2010_POINTS[(position - 1) as usize]
}

**1. Create `compute_championship_standings()` function:**
```rust
async fn compute_championship_standings(pool: &sqlx::SqlitePool, championship_id: &str) -> Vec<Value> {
```
- Query all rounds: `SELECT event_id, round_number FROM championship_rounds WHERE championship_id = ? ORDER BY round_number`
- If no rounds, return empty vec
- Compute standings with a single aggregation query:
```sql
SELECT hee.driver_id,
       COALESCE(d.nickname, d.name) as display_name,
       SUM(hee.points) as total_points,
       COUNT(DISTINCT cr.event_id) as rounds_entered,
       SUM(CASE WHEN hee.position = 1 AND hee.result_status = 'finished' THEN 1 ELSE 0 END) as wins,
       SUM(CASE WHEN hee.position = 2 AND hee.result_status = 'finished' THEN 1 ELSE 0 END) as p2_count,
       SUM(CASE WHEN hee.position = 3 AND hee.result_status = 'finished' THEN 1 ELSE 0 END) as p3_count,
       MIN(CASE WHEN hee.position <= 3 AND hee.result_status = 'finished' THEN hee.entered_at END) as earliest_podium,
       MIN(hee.position) as best_result
FROM hotlap_event_entries hee
INNER JOIN championship_rounds cr ON cr.event_id = hee.event_id
LEFT JOIN drivers d ON d.id = hee.driver_id
WHERE cr.championship_id = ?
  AND hee.result_status IN ('finished', 'dnf', 'dns')
GROUP BY hee.driver_id
ORDER BY total_points DESC, wins DESC, p2_count DESC, p3_count DESC, earliest_podium ASC
```
- Shape into Vec<Value> with position (1-indexed row number), driver_id, display_name, total_points, rounds_entered, wins, p2_count, p3_count, best_result
