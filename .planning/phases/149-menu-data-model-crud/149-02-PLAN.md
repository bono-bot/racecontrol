```typescript
// Cafe Menu
listCafeItems: () => fetchApi<{ items: CafeItem[]; total: number }>("/cafe/items"),
createCafeItem: (data: CreateCafeItemRequest) =>
  fetchApi<{ id: string }>("/cafe/items", { method: "POST", body: JSON.stringify(data) }),
updateCafeItem: (id: string, data: Partial<CreateCafeItemRequest>) =>
  fetchApi<CafeItem>(`/cafe/items/${id}`, { method: "PUT", body: JSON.stringify(data) }),
deleteCafeItem: (id: string) =>
  fetchApi<{ ok: boolean }>(`/cafe/items/${id}`, { method: "DELETE" }),
toggleCafeItem: (id: string) =>
  fetchApi<{ id: string; is_available: boolean }>(`/cafe/items/${id}/toggle`, { method: "POST" }),
listCafeCategories: () => fetchApi<{ categories: CafeCategory[] }>("/cafe/categories"),
createCafeCategory: (name: string, sort_order?: number) =>
  fetchApi<{ id: string; name: string }>("/cafe/categories", {
    method: "POST",
    body: JSON.stringify({ name, sort_order }),
  }),
```

Create `web/src/app/cafe/page.tsx` — a "use client" component wrapped in DashboardLayout.

**Page structure:**

```
DashboardLayout
  +-- Header: "Cafe Menu" title + "Add Item" button (rp-red)
  +-- Flex container (gap-6)
     +-- Main area (flex-1): Items table
     +-- Side panel (w-96, conditional): Add/Edit form
```

**State variables:**
- `items: CafeItem[]` — fetched on mount
- `categories: CafeCategory[]` — fetched on mount
- `loading: boolean` — true during initial fetch
- `showPanel: boolean` — controls side panel visibility
- `editItem: CafeItem | null` — null = add mode, non-null = edit mode
- `formData` — tracks form fields: name, description, category_id, selling_price_paise, cost_price_paise
- `newCategoryName: string` — for inline category creation
- `saving: boolean` — prevents double-submit

**Data loading (useEffect on mount):**
- Call `api.listCafeItems()` and `api.listCafeCategories()` in parallel
- Set items and categories state, set loading = false

**Items table columns:**
1. Name (text, sortable)
2. Category (display category name by looking up category_id in categories array)
3. Selling Price (format as rupees: `(paise / 100).toFixed(2)` with rupee prefix)
4. Cost Price (same format)
5. Status (green "Available" / red "Unavailable" badge)
6. Actions (Edit button, Toggle button, Delete button with confirmation)

**Table styling:** Use RP brand classes — `bg-rp-card`, `border-rp-border`, `text-white`, hover rows `hover:bg-rp-black/50`. Header row has `text-rp-grey text-xs uppercase`. Use `#E10600` for rp-red accent (buttons, active states). NOT the deprecated orange #FF4400.

**Side panel form (shown when showPanel = true):**
- Title: "Add Item" or "Edit Item" based on editItem state
- Fields:
  - Name (text input, required, placeholder "Item name")
  - Description (textarea, optional, placeholder "Brief description")
  - Category (select dropdown populated from categories array)
    - After the dropdown: a small "+" button that reveals an inline text input for new category name
    - On submit of new category: call `api.createCafeCategory(name)`, refresh categories, auto-select new category
  - Selling Price (number input, label "Selling Price (rupees)", value displayed as rupees but stored/sent as paise — multiply by 100 on submit)
  - Cost Price (number input, label "Cost Price (rupees)", same paise conversion)
- Buttons: "Save" (rp-red bg) + "Cancel" (neutral)
- On Save (add mode): call `api.createCafeItem(formData)`, refresh items list, close panel
- On Save (edit mode): call `api.updateCafeItem(editItem.id, formData)`, refresh items list, close panel

**Price conversion pattern:**
- Display: `(paise / 100).toFixed(2)` — shows "149.50" for 14950 paise
- Form input: user types rupees (e.g., "149.50"), on submit convert: `Math.round(parseFloat(value) * 100)`
- This matches the billing/pricing page pattern

**Action handlers:**
- Edit: set editItem to clicked item, populate formData from item values (converting paise to rupees for display), set showPanel = true
- Toggle: call `api.toggleCafeItem(id)`, update item in local state (optimistic or refetch)
- Delete: `window.confirm("Delete {item.name}?")` then call `api.deleteCafeItem(id)` then remove from local state
- Add: set editItem = null, clear formData, set showPanel = true

**Empty state:** When items array is empty after loading, show "No cafe items yet. Click 'Add Item' to get started." centered text.

**Loading state:** Show "Loading..." text while fetching.

**Error handling:** Wrap all API calls in try/catch. On error, show `alert(error.message)` (simple for now — toast system can be added later).

**CRITICAL: No sessionStorage/localStorage in useState initializer** — per CLAUDE.md, use useEffect for any storage reads.
