```javascript
/**
 * Full failback orchestration sequence — reverse of initiateFailover().
 * Triggered by HealthMonitor 'server_recovery' event (down -> healthy).
 *
 * Steps:
 *   1. Cooldown check (same alertCooldown as failover)
 *   2. 30s stabilization wait
 *   3. Re-probe .23 to confirm still up
 *   4. Send export_failover_sessions exec_request to Bono
 *   5. POST exported sessions to .23 /api/v1/sync/import-sessions
 *   6. Broadcast SwitchController with local .23 URL via cloud racecontrol
 *   7. Wait 30s for pods to reconnect
 *   8. Deactivate cloud racecontrol (pm2 stop)
 *   9. Notify Uday with all-clear + outage duration
 *
 * @returns {Promise<void>}
 */
async initiateFailback() {
  // Step 1: Cooldown check
  if (!this.#alertCooldown.canSend()) {
    console.log('[FAILBACK] Suppressed by cooldown — skipping');
    return;
  }

  const recoveryTimestamp = Date.now();
  console.log('[FAILBACK] Initiating — server .23 recovery detected');

  // Step 2: 30s stabilization
  console.log('[FAILBACK] Waiting 30s stabilization period...');
  await sleep(30_000);

  // Step 3: Re-probe .23
  const stillUp = await this.#httpGet('http://192.168.31.23:8080/api/v1/health');
  if (!stillUp) {
    console.warn('[FAILBACK] Server .23 went back down during stabilization — aborting failback');
    return;
  }
  console.log('[FAILBACK] Server .23 confirmed still up after stabilization');

  // Step 4: Export failover sessions from Bono
  let sessionCount = 0;
  let syncError = null;
  const failoverStart = this.#failoverStartedAt || (recoveryTimestamp - 3_600_000); // fallback: 1h ago
  const failoverStartIso = new Date(failoverStart).toISOString();

  const exportExecId = `ex_${randomUUID().slice(0, 8)}`;
  this.#client.send('exec_request', {
    execId: exportExecId,
    command: 'export_failover_sessions',
    reason: `failback: export sessions since ${failoverStartIso}`,
    requestedBy: 'james',
  });
  console.log(`[FAILBACK] Sent export_failover_sessions (execId=${exportExecId})`);

  const exportResult = await this.#waitForExecResult(exportExecId, 30_000).catch((err) => {
    console.warn(`[FAILBACK] export_failover_sessions timeout/error: ${err.message}`);
    return null;
  });

  // Step 5: Import sessions to .23
  if (exportResult && exportResult.exitCode === 0 && exportResult.stdout) {
    try {
      let allSessions = JSON.parse(exportResult.stdout);
      // Filter to sessions created during failover window
      const sessions = allSessions.filter((s) => s.created_at >= failoverStartIso);
      sessionCount = sessions.length;

      if (sessions.length > 0) {
        const TERMINAL_SECRET = process.env.TERMINAL_SECRET || 'rp-terminal-2026';
        const importResp = await this.#httpPost(
          'http://192.168.31.23:8080/api/v1/sync/import-sessions',
          JSON.stringify({ sessions }),
          { 'x-terminal-secret': TERMINAL_SECRET, 'Content-Type': 'application/json' },
        );
        console.log(`[FAILBACK] import-sessions response: status=${importResp.statusCode} body=${importResp.body}`);
      } else {
        console.log('[FAILBACK] No sessions to import from failover window');
      }
    } catch (err) {
      syncError = err.message;
      console.error(`[FAILBACK] Session import failed: ${err.message}`);
    }
  } else {
    syncError = exportResult ? `exitCode=${exportResult.exitCode}` : 'no response';
    console.warn(`[FAILBACK] Session export failed (${syncError}) — proceeding with pod switchback`);
  }

  // Step 6: Broadcast SwitchController back to .23 (via cloud racecontrol)
  // First, verify cloud racecontrol is running (it might have been manually stopped)
  const cloudHealthExecId = `ex_${randomUUID().slice(0, 8)}`;
  this.#client.send('exec_request', {
    execId: cloudHealthExecId,
    command: 'racecontrol_health',
    reason: 'failback: verify cloud racecontrol is up before broadcast',
    requestedBy: 'james',
  });
  const cloudHealthResult = await this.#waitForExecResult(cloudHealthExecId, 10_000).catch(() => null);

  if (!cloudHealthResult || cloudHealthResult.exitCode !== 0) {
    // Cloud racecontrol might be stopped — try to start it for the broadcast
    console.log('[FAILBACK] Cloud racecontrol not responding — activating for broadcast');
    const activateExecId = `ex_${randomUUID().slice(0, 8)}`;
    this.#client.send('exec_request', {
      execId: activateExecId,
      command: 'activate_failover',
      reason: 'failback: temporarily reactivate cloud for broadcast',
      requestedBy: 'james',
    });
    await this.#waitForExecResult(activateExecId, 15_000).catch(() => null);
    await sleep(5_000); // Give it time to start
  }

  const TERMINAL_SECRET = process.env.TERMINAL_SECRET || 'rp-terminal-2026';
  let broadcastResult = null;

  for (let attempt = 1; attempt <= 3; attempt++) {
    try {
      console.log(`[FAILBACK] Broadcasting SwitchController back to .23, attempt ${attempt}/3`);
      const resp = await this.#httpPost(
        'http://100.70.177.44:8080/api/v1/failover/broadcast',
        JSON.stringify({ target_url: 'ws://192.168.31.23:8080/ws/agent' }),
        { 'x-terminal-secret': TERMINAL_SECRET },
      );

      if (resp.statusCode >= 200 && resp.statusCode < 300) {
        console.log(`[FAILBACK] Broadcast successful (status=${resp.statusCode})`);
        try { broadcastResult = JSON.parse(resp.body); } catch (_) {}
        break;
      }
      console.warn(`[FAILBACK] Broadcast returned ${resp.statusCode}: ${resp.body}`);
    } catch (err) {
      console.warn(`[FAILBACK] Broadcast error (attempt ${attempt}/3): ${err.message}`);
    }

    if (attempt < 3) {
      console.log(`[FAILBACK] Waiting 5s before broadcast retry ${attempt + 1}/3`);
      await sleep(5_000);
    }
  }

  if (!broadcastResult) {
    console.error('[FAILBACK] Broadcast failed after 3 attempts — alerting Uday for manual intervention');
  }

  // Step 7: Wait 30s for pods to reconnect
  console.log('[FAILBACK] Waiting 30s for pods to reconnect to .23...');
  await sleep(30_000);

  // Step 8: Deactivate cloud racecontrol
  const deactivateExecId = `ex_${randomUUID().slice(0, 8)}`;
  this.#client.send('exec_request', {
    execId: deactivateExecId,
    command: 'deactivate_failover',
    reason: 'failback: pods have reconnected to .23',
    requestedBy: 'james',
  });
  console.log(`[FAILBACK] Sent deactivate_failover (execId=${deactivateExecId})`);
  await this.#waitForExecResult(deactivateExecId, 30_000).catch(() => null);

  // Step 9: Notify Uday — all-clear + outage duration
  this.#alertCooldown.recordSent();

  const outageMs = recoveryTimestamp - (this.#failoverStartedAt || recoveryTimestamp);
  const totalMinutes = Math.round(outageMs / 60_000);
  const hours = Math.floor(totalMinutes / 60);
  const minutes = totalMinutes % 60;
  const outageFormatted = `${hours}h ${minutes}m`;
  const podsSent = broadcastResult?.sent ?? '?';
  const podsTotal = broadcastResult?.total ?? '8';

  const notifyReason = [
    `FAILBACK COMPLETE — Server .23 recovered.`,
    `Outage duration: ${outageFormatted}.`,
    `Sessions synced: ${sessionCount}.`,
    syncError ? `Sync warning: ${syncError}.` : '',
    `Pods switched: ${podsSent}/${podsTotal}.`,
    `All pods back on local server.`,
    `Time: ${istTimestamp()}.`,
  ].filter(Boolean).join(' ');

  const notifyExecId = `ex_${randomUUID().slice(0, 8)}`;
  this.#client.send('exec_request', {
    execId: notifyExecId,
    command: 'notify_failback',
    reason: notifyReason,
    requestedBy: 'james',
  });
  console.log(`[FAILBACK] Sent notify_failback to Bono (execId=${notifyExecId})`);

  // Email notification (fire-and-forget)
  try {
    const { execFile } = await import('node:child_process');
    const emailPath = process.env.SEND_EMAIL_PATH || new URL('../shared/send-email.js', import.meta.url).pathname;
    execFile('node', [emailPath, 'usingh@racingpoint.in', 'FAILBACK COMPLETE', notifyReason], { timeout: 30000 }, (err) => {
      if (err) console.error('[FAILBACK] Email notification failed:', err.message);
      else console.log('[FAILBACK] Failback email sent to Uday');
    });
  } catch (err) {
    console.error('[FAILBACK] Email setup failed:', err.message);
  }

  // Reset failover timestamp
  this.#failoverStartedAt = null;

  console.log(`[FAILBACK] Orchestration complete — ${notifyReason}`);
}
```

/**
 * Simple HTTP GET with 5s timeout. Returns true on 2xx response.
 * @param {string} url
 * @returns {Promise<boolean>}
 * @private
 */
#httpGet(url) {
  return new Promise((resolve) => {
    const http = require('node:http');
    const req = http.get(url, { timeout: 5_000 }, (res) => {
      resolve(res.statusCode >= 200 && res.statusCode < 300);
      res.resume();
    });
    req.on('error', () => resolve(false));
    req.on('timeout', () => { req.destroy(); resolve(false); });
  });
}
